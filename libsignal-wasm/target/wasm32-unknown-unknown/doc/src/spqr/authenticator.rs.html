4use libcrux_hmac::hmac;
5
6use crate::{kdf, util::compare, Epoch};
7pub mod serialize;
8pub type Mac = Vec<u8>;
9
10#[derive(Debug, thiserror::Error)]
11pub enum Error {
12    #[error("Ciphertext MAC is invalid")]
13    InvalidCtMac,
14    #[error("Encapsulation key MAC is invalid")]
15    InvalidHdrMac,
16    #[error("Authenticator previous root key present when should be erased")]
17    AuthenticatorRootKeyPresent,
18    #[error("Authenticator previous root key missing")]
19    AuthenticatorRootKeyMissing,
20    #[error("Authenticator previous MAC key present when should be erased")]
21    AuthenticatorMacKeyPresent,
22    #[error("Authenticator previous MAC key missing")]
23    AuthenticatorMacKeyMissing,
24}
25
26#[cfg_attr(test, derive(Clone))]
27pub struct Authenticator {
28    root_key: Mac,
29    mac_key: Mac,
30}
31
32#[hax_lib::attributes]
33impl Authenticator {
34    pub const MACSIZE: usize = 32usize;
35    pub fn new(root_key: Vec<u8>, ep: Epoch) -> Self {
36        let mut result = Self {
37            root_key: vec![0u8; 32],
38            mac_key: vec![0u8; 32],
39        };
40        result.update(ep, &root_key);
41        result
42    }
43
44    pub fn update(&mut self, ep: Epoch, k: &[u8]) {
45        let ikm = [self.root_key.as_slice(), k].concat();
46        let info = [
47            b"Signal_PQCKA_V1_MLKEM768:Authenticator Update".as_slice(),
48            &ep.to_be_bytes(),
49        ]
50        .concat();
51        let kdf_out = kdf::hkdf_to_vec(&[0u8; 32], &ikm, &info, 64);
52        self.root_key = kdf_out[..32].to_vec();
53        self.mac_key = kdf_out[32..].to_vec();
54    }
55
56    #[hax_lib::requires(expected_mac.len() == Authenticator::MACSIZE)]
57    pub fn verify_ct(&self, ep: Epoch, ct: &[u8], expected_mac: &[u8]) -> Result<(), Error> {
58        if compare(expected_mac, &self.mac_ct(ep, ct)) != 0 {
59            Err(Error::InvalidCtMac)
60        } else {
61            Ok(())
62        }
63    }
64
65    #[hax_lib::ensures(|res| res.len() == Authenticator::MACSIZE)]
66    pub fn mac_ct(&self, ep: Epoch, ct: &[u8]) -> Mac {
67        let ct_mac_data = [
68            b"Signal_PQCKA_V1_MLKEM768:ciphertext".as_slice(),
69            &ep.to_be_bytes(),
70            ct,
71        ]
72        .concat();
73        hmac(
74            libcrux_hmac::Algorithm::Sha256,
75            &self.mac_key,
76            &ct_mac_data,
77            Some(Self::MACSIZE),
78        )
79    }
80
81    #[hax_lib::requires(expected_mac.len() == Authenticator::MACSIZE)]
82    pub fn verify_hdr(&self, ep: Epoch, hdr: &[u8], expected_mac: &[u8]) -> Result<(), Error> {
83        if compare(expected_mac, &self.mac_hdr(ep, hdr)) != 0 {
84            Err(Error::InvalidHdrMac)
85        } else {
86            Ok(())
87        }
88    }
89
90    #[hax_lib::ensures(|res| res.len() == Authenticator::MACSIZE)]
91    pub fn mac_hdr(&self, ep: Epoch, hdr: &[u8]) -> Mac {
92        let ct_mac_data = [
93            b"Signal_PQCKA_V1_MLKEM768:ekheader".as_slice(),
94            &ep.to_be_bytes(),
95            hdr,
96        ]
97        .concat();
98        hmac(
99            libcrux_hmac::Algorithm::Sha256,
100            &self.mac_key,
101            &ct_mac_data,
102            Some(Self::MACSIZE),
103        )
104    }
105}
