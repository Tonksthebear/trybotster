22pub struct Keys {

28pub fn ek_matches_header(ek: &EncapsulationKey, hdr: &Header) -> bool {
29    incremental::validate_pk_bytes(hdr, ek).is_ok()
30}
31
32/// Generate a new keypair and associated header.
33#[hax_lib::ensures(|result| result.hdr.len() == 64 && result.ek.len() == 1152 && result.dk.len() == 2400)]
34pub fn generate<R: Rng + CryptoRng>(rng: &mut R) -> Keys {
35    let mut randomness = [0u8; libcrux_ml_kem::KEY_GENERATION_SEED_SIZE];
36    rng.fill_bytes(&mut randomness);
37    let k = incremental::KeyPairCompressedBytes::from_seed(randomness);
38    Keys {
39        hdr: k.pk1().to_vec(),
40        ek: k.pk2().to_vec(),
41        dk: k.sk().to_vec(),
42    }
43}
44
45/// Encapsulate with header to get initial ciphertext.
46#[hax_lib::requires(hdr.len() == 64)]
47#[hax_lib::ensures(|(ct1,es,ss)| ct1.len() == 960 && es.len() == 2080 && ss.len() == 32)]
48pub fn encaps1<R: Rng + CryptoRng>(
49    hdr: &Header,
50    rng: &mut R,
51) -> (Ciphertext1, EncapsulationState, Secret) {
52    let mut randomness = [0u8; libcrux_ml_kem::SHARED_SECRET_SIZE];
53    rng.fill_bytes(&mut randomness);
54    let mut state = vec![0u8; incremental::encaps_state_len()];
55    let mut ss = vec![0u8; libcrux_ml_kem::SHARED_SECRET_SIZE];
56    let ct1 = incremental::encapsulate1(hdr.as_slice(), randomness, &mut state, &mut ss);
57    hax_lib::assume!(ct1.is_ok());
58    hax_lib::assume!(state.len() == 2080 && ss.len() == 32);
59    (
60        ct1.expect("should only fail based on sizes, all sizes should be correct")
61            .value
62            .to_vec(),
63        state,
64        ss,
65    )
66}
67
68/// Encapsulate with header and EK.
69#[hax_lib::requires(es.len() == 2080 && ek.len() == 1152)]
70#[hax_lib::ensures(|result| result.len() == 128)]
71pub fn encaps2(ek: &EncapsulationKey, es: &EncapsulationState) -> Ciphertext2 {
72    let ct2 = incremental::encapsulate2(
73        es.as_slice().try_into().expect("size should be correct"),
74        ek.as_slice().try_into().expect("size should be correct"),
75    );
76    ct2.value.to_vec()
77}
78
79/// Decapsulate ciphertext to get shared secret.
80#[hax_lib::requires(ct1.len() == 960 && ct2.len() == 128 && dk.len() == 2400)]
81#[hax_lib::ensures(|result| result.len() == 32)]
82pub fn decaps(dk: &DecapsulationKey, ct1: &Ciphertext1, ct2: &Ciphertext2) -> Secret {
83    let ct1 = incremental::Ciphertext1 {
84        value: ct1.as_slice().try_into().expect("size should be correct"),
85    };
86    let ct2 = incremental::Ciphertext2 {
87        value: ct2.as_slice().try_into().expect("size should be correct"),
88    };
89    incremental::decapsulate_compressed_key(
90        dk.as_slice().try_into().expect("size should be correct"),
91        &ct1,
92        &ct2,
93    )
94    .to_vec()
95}

97#[cfg(test)]
98mod test {
99    use super::*;
100    use rand::TryRngCore;
101    use rand_core::OsRng;
102
103    #[test]
104    fn incremental_mlkem768_round_trip() {
105        let mut rng = OsRng.unwrap_err();
106        let keys = generate(&mut rng);
107        let (ct1, es, ss1) = encaps1(&keys.hdr, &mut rng);
108        let ct2 = encaps2(&keys.ek, &es);
109        let ss2 = decaps(&keys.dk, &ct1, &ct2);
110        assert_eq!(ss1, ss2);
111    }
112}
