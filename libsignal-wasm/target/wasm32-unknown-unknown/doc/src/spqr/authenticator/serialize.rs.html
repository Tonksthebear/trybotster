4use crate::proto;
5
6use super::Authenticator;
7
8impl Authenticator {
9    pub fn into_pb(self) -> proto::pq_ratchet::Authenticator {
10        proto::pq_ratchet::Authenticator {
11            root_key: self.root_key,
12            mac_key: self.mac_key,
13        }
14    }
15
16    pub fn from_pb(pb: &proto::pq_ratchet::Authenticator) -> Self {
17        Self {
18            root_key: pb.root_key.clone(),
19            mac_key: pb.mac_key.clone(),
20        }
21    }
22}
23
24#[cfg(test)]
25mod test {
26    use crate::authenticator::Authenticator;
27
28    #[test]
29    fn round_trip() {
30        let auth = Authenticator::new(vec![42u8; 32], 1);
31        let ack = auth.mac_ct(1, b"123");
32
33        let pb_auth = auth.into_pb();
34
35        let new_auth = Authenticator::from_pb(&pb_auth);
36        let new_mac = new_auth.mac_ct(1, b"123");
37
38        assert_eq!(ack, new_mac);
39    }
40}
