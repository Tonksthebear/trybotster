1//! This module contains the trait and related errors for an Authenticated
2//! Encryption with Associated Data (AEAD) scheme that takes array references
3//! as arguments and returns outputs as arrays or vectors.
4
5use super::arrayref::{DecryptError, EncryptError, KeyGenError};
6use libcrux_secrets::{Classify, U8};
7
8/// An Authenticated Encryption with Associated Data (AEAD) scheme
9///
10/// Some implementors of this trait may impose stronger restrictions on the inputs than described
11/// here. Check the documentation of the types implementing this trait to make sure which inputs
12/// are valid.
13pub trait Aead<const KEY_LEN: usize, const TAG_LEN: usize, const NONCE_LEN: usize> {
14    /// Generate a new key. Consumes the entire randomnes.
15    fn keygen(rand: &[U8; KEY_LEN]) -> Result<[U8; KEY_LEN], KeyGenError>;
16
17    /// Encrypt a plaintext message, producing a ciphertext and an authentication tag.
18    fn encrypt<const MSG_LEN: usize>(
19        key: &[U8; KEY_LEN],
20        nonce: &[U8; NONCE_LEN],
21        aad: &[u8],
22        plaintext: &[U8; MSG_LEN],
23    ) -> Result<([u8; MSG_LEN], [U8; TAG_LEN]), EncryptError>;
24
25    /// Encrypt a plaintext message, producing a ciphertext and an authentication tag.
26    #[cfg(feature = "alloc")]
27    fn encrypt_to_vec(
28        key: &[u8; KEY_LEN],
29        nonce: &[u8; NONCE_LEN],
30        aad: &[u8],
31        plaintext: &[u8],
32    ) -> Result<(alloc::vec::Vec<u8>, [u8; TAG_LEN]), EncryptError>;
33
34    /// Decrypt a ciphertext, verifying its authenticity.
35    fn decrypt<const MSG_LEN: usize>(
36        key: &[U8; KEY_LEN],
37        nonce: &[U8; NONCE_LEN],
38        aad: &[u8],
39        ciphertext: &[u8; MSG_LEN],
40        tag: &[U8; TAG_LEN],
41    ) -> Result<[U8; MSG_LEN], DecryptError>;
42
43    /// Decrypt a ciphertext, verifying its authenticity.
44    #[cfg(feature = "alloc")]
45    fn decrypt_to_vec(
46        key: &[U8; KEY_LEN],
47        nonce: &[u8; NONCE_LEN],
48        aad: &[u8],
49        ciphertext: &[u8],
50        tag: &[u8; TAG_LEN],
51    ) -> Result<alloc::vec::Vec<U8>, DecryptError>;
52}
53
54impl<
55        const KEY_LEN: usize,
56        const TAG_LEN: usize,
57        const NONCE_LEN: usize,
58        T: super::arrayref::Aead<KEY_LEN, TAG_LEN, NONCE_LEN>,
59    > Aead<KEY_LEN, TAG_LEN, NONCE_LEN> for T
60{
61    fn keygen(rand: &[U8; KEY_LEN]) -> Result<[U8; KEY_LEN], KeyGenError> {
62        let mut key = [0u8.classify(); KEY_LEN];
63        Self::keygen(&mut key, rand)?;
64        Ok(key)
65    }
66
67    fn encrypt<const MSG_LEN: usize>(
68        key: &[U8; KEY_LEN],
69        nonce: &[U8; NONCE_LEN],
70        aad: &[u8],
71        plaintext: &[U8; MSG_LEN],
72    ) -> Result<([u8; MSG_LEN], [U8; TAG_LEN]), EncryptError> {
73        let mut ciphertext = [0u8; MSG_LEN];
74        let mut tag = [0u8.classify(); TAG_LEN];
75
76        Self::encrypt(&mut ciphertext, &mut tag, key, nonce, aad, plaintext)
77            .map(|()| (ciphertext, tag))
78    }
79
80    #[cfg(feature = "alloc")]
81    fn encrypt_to_vec(
82        key: &[U8; KEY_LEN],
83        nonce: &[u8; NONCE_LEN],
84        aad: &[u8],
85        plaintext: &[U8],
86    ) -> Result<(alloc::vec::Vec<u8>, [u8; TAG_LEN]), EncryptError> {
87        let mut ciphertext = alloc::vec![0u8; plaintext.len()];
88        let mut tag = [0u8; TAG_LEN];
89
90        Self::encrypt(&mut ciphertext, &mut tag, key, nonce, aad, plaintext)
91            .map(|()| (ciphertext, tag))
92    }
93
94    fn decrypt<const MSG_LEN: usize>(
95        key: &[U8; KEY_LEN],
96        nonce: &[U8; NONCE_LEN],
97        aad: &[u8],
98        ciphertext: &[u8; MSG_LEN],
99        tag: &[U8; TAG_LEN],
100    ) -> Result<[U8; MSG_LEN], DecryptError> {
101        let mut plaintext = [0u8.classify(); MSG_LEN];
102
103        Self::decrypt(&mut plaintext, key, nonce, aad, ciphertext, tag).map(|()| plaintext)
104    }
105
106    #[cfg(feature = "alloc")]
107    fn decrypt_to_vec(
108        key: &[U8; KEY_LEN],
109        nonce: &[u8; NONCE_LEN],
110        aad: &[u8],
111        ciphertext: &[u8],
112        tag: &[u8; TAG_LEN],
113    ) -> Result<alloc::vec::Vec<U8>, DecryptError> {
114        let mut plaintext = alloc::vec![0u8; ciphertext.len()];
115
116        Self::decrypt(&mut plaintext, key, nonce, aad, ciphertext, tag).map(|()| plaintext)
117    }
118}
