1//! Arithmetic mod 2^252 + 27742317777372353535851937790883648493
2//! with 9 29-bit unsigned limbs
3//!
4//! To see that this is safe for intermediate results, note that
5//! the largest limb in a 9 by 9 product of 29-bit limbs will be
6//! (0x1fffffff^2) * 9 = 0x23fffffdc0000009 (62 bits).
7//!
8//! For a one level Karatsuba decomposition, the specific ranges
9//! depend on how the limbs are combined, but will stay within
10//! -0x1ffffffe00000008 (62 bits with sign bit) to
11//! 0x43fffffbc0000011 (63 bits), which is still safe.
12

21/// The `Scalar29` struct represents an element in \\(\mathbb{Z} / \ell\mathbb{Z}\\) as 9 29-bit
22/// limbs
23#[derive(Copy, Clone)]
24pub struct Scalar29(pub [u32; 9]);

62    /// Unpack a 32 byte / 256 bit scalar into 9 29-bit limbs.
63    #[rustfmt::skip] // keep alignment of s[*] calculations
64    pub fn from_bytes(bytes: &[u8; 32]) -> Scalar29 {
65        let mut words = [0u32; 8];
66        for i in 0..8 {
67            for j in 0..4 {
68                words[i] |= (bytes[(i * 4) + j] as u32) << (j * 8);
69            }
70        }
71
72        let mask = (1u32 << 29) - 1;
73        let top_mask = (1u32 << 24) - 1;
74        let mut s = Scalar29::ZERO;
75
76        s[0] =   words[0]                            & mask;
77        s[1] = ((words[0] >> 29) | (words[1] <<  3)) & mask;
78        s[2] = ((words[1] >> 26) | (words[2] <<  6)) & mask;
79        s[3] = ((words[2] >> 23) | (words[3] <<  9)) & mask;
80        s[4] = ((words[3] >> 20) | (words[4] << 12)) & mask;
81        s[5] = ((words[4] >> 17) | (words[5] << 15)) & mask;
82        s[6] = ((words[5] >> 14) | (words[6] << 18)) & mask;
83        s[7] = ((words[6] >> 11) | (words[7] << 21)) & mask;
84        s[8] =  (words[7] >>  8)                     & top_mask;
85
86        s
87    }

89    /// Reduce a 64 byte / 512 bit scalar mod l.
90    #[rustfmt::skip] // keep alignment of lo[*] calculations
91    pub fn from_bytes_wide(bytes: &[u8; 64]) -> Scalar29 {
92        let mut words = [0u32; 16];
93        for i in 0..16 {
94            for j in 0..4 {
95                words[i] |= (bytes[(i * 4) + j] as u32) << (j * 8);
96            }
97        }
98
99        let mask = (1u32 << 29) - 1;
100        let mut lo = Scalar29::ZERO;
101        let mut hi = Scalar29::ZERO;
102
103        lo[0] =   words[ 0]                             & mask;
104        lo[1] = ((words[ 0] >> 29) | (words[ 1] <<  3)) & mask;
105        lo[2] = ((words[ 1] >> 26) | (words[ 2] <<  6)) & mask;
106        lo[3] = ((words[ 2] >> 23) | (words[ 3] <<  9)) & mask;
107        lo[4] = ((words[ 3] >> 20) | (words[ 4] << 12)) & mask;
108        lo[5] = ((words[ 4] >> 17) | (words[ 5] << 15)) & mask;
109        lo[6] = ((words[ 5] >> 14) | (words[ 6] << 18)) & mask;
110        lo[7] = ((words[ 6] >> 11) | (words[ 7] << 21)) & mask;
111        lo[8] = ((words[ 7] >>  8) | (words[ 8] << 24)) & mask;
112        hi[0] = ((words[ 8] >>  5) | (words[ 9] << 27)) & mask;
113        hi[1] =  (words[ 9] >>  2)                      & mask;
114        hi[2] = ((words[ 9] >> 31) | (words[10] <<  1)) & mask;
115        hi[3] = ((words[10] >> 28) | (words[11] <<  4)) & mask;
116        hi[4] = ((words[11] >> 25) | (words[12] <<  7)) & mask;
117        hi[5] = ((words[12] >> 22) | (words[13] << 10)) & mask;
118        hi[6] = ((words[13] >> 19) | (words[14] << 13)) & mask;
119        hi[7] = ((words[14] >> 16) | (words[15] << 16)) & mask;
120        hi[8] =   words[15] >> 13                             ;
121
122        lo = Scalar29::montgomery_mul(&lo, &constants::R);  // (lo * R) / R = lo
123        hi = Scalar29::montgomery_mul(&hi, &constants::RR); // (hi * R^2) / R = hi * R
124
125        Scalar29::add(&hi, &lo) // (hi * R) + lo
126    }

170    /// Compute `a + b` (mod l).
171    pub fn add(a: &Scalar29, b: &Scalar29) -> Scalar29 {
172        let mut sum = Scalar29::ZERO;
173        let mask = (1u32 << 29) - 1;
174
175        // a + b
176        let mut carry: u32 = 0;
177        for i in 0..9 {
178            carry = a[i] + b[i] + (carry >> 29);
179            sum[i] = carry & mask;
180        }
181
182        // subtract l if the sum is >= l
183        Scalar29::sub(&sum, &constants::L)
184    }

186    /// Compute `a - b` (mod l).
187    pub fn sub(a: &Scalar29, b: &Scalar29) -> Scalar29 {
188        // Optimization barrier to prevent compiler from inserting branch instructions
189        // TODO(tarcieri): find a better home (or abstraction) for this
190        fn black_box(value: u32) -> u32 {
191            // SAFETY: `u32` is a simple integer `Copy` type and `value` lives on the stack so
192            // a pointer to it will be valid.
193            unsafe { core::ptr::read_volatile(&value) }
194        }
195
196        let mut difference = Scalar29::ZERO;
197        let mask = (1u32 << 29) - 1;
198
199        // a - b
200        let mut borrow: u32 = 0;
201        for i in 0..9 {
202            borrow = a[i].wrapping_sub(b[i] + (borrow >> 31));
203            difference[i] = borrow & mask;
204        }
205
206        // conditionally add l if the difference is negative
207        let underflow_mask = ((borrow >> 31) ^ 1).wrapping_sub(1);
208        let mut carry: u32 = 0;
209        for i in 0..9 {
210            carry = (carry >> 29) + difference[i] + (constants::L[i] & black_box(underflow_mask));
211            difference[i] = carry & mask;
212        }
213
214        difference
215    }

217    /// Compute `a * b`.
218    ///
219    /// This is implemented with a one-level refined Karatsuba decomposition
220    #[inline(always)]
221    #[rustfmt::skip] // keep alignment of z[*] calculations
222    pub (crate) fn mul_internal(a: &Scalar29, b: &Scalar29) -> [u64; 17] {
223        let mut z = [0u64; 17];
224
225        z[0] = m(a[0], b[0]);                                                                 // c00
226        z[1] = m(a[0], b[1]) + m(a[1], b[0]);                                                 // c01
227        z[2] = m(a[0], b[2]) + m(a[1], b[1]) + m(a[2], b[0]);                                 // c02
228        z[3] = m(a[0], b[3]) + m(a[1], b[2]) + m(a[2], b[1]) + m(a[3], b[0]);                 // c03
229        z[4] = m(a[0], b[4]) + m(a[1], b[3]) + m(a[2], b[2]) + m(a[3], b[1]) + m(a[4], b[0]); // c04
230        z[5] =                 m(a[1], b[4]) + m(a[2], b[3]) + m(a[3], b[2]) + m(a[4], b[1]); // c05
231        z[6] =                                 m(a[2], b[4]) + m(a[3], b[3]) + m(a[4], b[2]); // c06
232        z[7] =                                                 m(a[3], b[4]) + m(a[4], b[3]); // c07
233        z[8] =                                                                (m(a[4], b[4])).wrapping_sub(z[3]); // c08 - c03
234
235        z[10] = z[5].wrapping_sub(m(a[5], b[5]));                                                // c05mc10
236        z[11] = z[6].wrapping_sub(m(a[5], b[6]) + m(a[6], b[5]));                                // c06mc11
237        z[12] = z[7].wrapping_sub(m(a[5], b[7]) + m(a[6], b[6]) + m(a[7], b[5]));                // c07mc12
238        z[13] =                   m(a[5], b[8]) + m(a[6], b[7]) + m(a[7], b[6]) + m(a[8], b[5]); // c13
239        z[14] =                                   m(a[6], b[8]) + m(a[7], b[7]) + m(a[8], b[6]); // c14
240        z[15] =                                                   m(a[7], b[8]) + m(a[8], b[7]); // c15
241        z[16] =                                                                   m(a[8], b[8]); // c16
242
243        z[ 5] = z[10].wrapping_sub(z[ 0]); // c05mc10 - c00
244        z[ 6] = z[11].wrapping_sub(z[ 1]); // c06mc11 - c01
245        z[ 7] = z[12].wrapping_sub(z[ 2]); // c07mc12 - c02
246        z[ 8] = z[ 8].wrapping_sub(z[13]); // c08mc13 - c03
247        z[ 9] = z[14].wrapping_add(z[ 4]); // c14 + c04
248        z[10] = z[15].wrapping_add(z[10]); // c15 + c05mc10
249        z[11] = z[16].wrapping_add(z[11]); // c16 + c06mc11
250
251        let aa = [
252            a[0] + a[5],
253            a[1] + a[6],
254            a[2] + a[7],
255            a[3] + a[8]
256        ];
257
258        let bb = [
259            b[0] + b[5],
260            b[1] + b[6],
261            b[2] + b[7],
262            b[3] + b[8]
263        ];
264
265        z[ 5] = (m(aa[0], bb[0]))                                                                       .wrapping_add(z[ 5]); // c20 + c05mc10 - c00
266        z[ 6] = (m(aa[0], bb[1]) + m(aa[1], bb[0]))                                                     .wrapping_add(z[ 6]); // c21 + c06mc11 - c01
267        z[ 7] = (m(aa[0], bb[2]) + m(aa[1], bb[1]) + m(aa[2], bb[0]))                                   .wrapping_add(z[ 7]); // c22 + c07mc12 - c02
268        z[ 8] = (m(aa[0], bb[3]) + m(aa[1], bb[2]) + m(aa[2], bb[1]) + m(aa[3], bb[0]))                 .wrapping_add(z[ 8]); // c23 + c08mc13 - c03
269        z[ 9] = (m(aa[0],  b[4]) + m(aa[1], bb[3]) + m(aa[2], bb[2]) + m(aa[3], bb[1]) + m(a[4], bb[0])).wrapping_sub(z[ 9]); // c24 - c14 - c04
270        z[10] = (                  m(aa[1],  b[4]) + m(aa[2], bb[3]) + m(aa[3], bb[2]) + m(a[4], bb[1])).wrapping_sub(z[10]); // c25 - c15 - c05mc10
271        z[11] = (                                    m(aa[2],  b[4]) + m(aa[3], bb[3]) + m(a[4], bb[2])).wrapping_sub(z[11]); // c26 - c16 - c06mc11
272        z[12] = (                                                      m(aa[3],  b[4]) + m(a[4], bb[3])).wrapping_sub(z[12]); // c27 - c07mc12
273
274        z
275    }

280    fn square_internal(a: &Scalar29) -> [u64; 17] {

284            a[2] * 2,

292        [

294            m(aa[0], a[1]),

313    /// Compute `limbs/R` (mod l), where R is the Montgomery modulus 2^261
314    #[inline(always)]
315    #[rustfmt::skip] // keep alignment of part1() and part2() computations
316    pub (crate) fn montgomery_reduce(limbs: &[u64; 17]) -> Scalar29 {
317
318        #[inline(always)]
319        fn part1(sum: u64) -> (u64, u32) {
320            let p = (sum as u32).wrapping_mul(constants::LFACTOR) & ((1u32 << 29) - 1);
321            ((sum + m(p,constants::L[0])) >> 29, p)
322        }
323
324        #[inline(always)]
325        fn part2(sum: u64) -> (u64, u32) {
326            let w = (sum as u32) & ((1u32 << 29) - 1);
327            (sum >> 29, w)
328        }
329
330        // note: l5,l6,l7 are zero, so their multiplies can be skipped
331        let l = &constants::L;
332
333        // the first half computes the Montgomery adjustment factor n, and begins adding n*l to make limbs divisible by R
334        let (carry, n0) = part1(        limbs[ 0]);
335        let (carry, n1) = part1(carry + limbs[ 1] + m(n0,l[1]));
336        let (carry, n2) = part1(carry + limbs[ 2] + m(n0,l[2]) + m(n1,l[1]));
337        let (carry, n3) = part1(carry + limbs[ 3] + m(n0,l[3]) + m(n1,l[2]) + m(n2,l[1]));
338        let (carry, n4) = part1(carry + limbs[ 4] + m(n0,l[4]) + m(n1,l[3]) + m(n2,l[2]) + m(n3,l[1]));
339        let (carry, n5) = part1(carry + limbs[ 5]              + m(n1,l[4]) + m(n2,l[3]) + m(n3,l[2]) + m(n4,l[1]));
340        let (carry, n6) = part1(carry + limbs[ 6]                           + m(n2,l[4]) + m(n3,l[3]) + m(n4,l[2]) + m(n5,l[1]));
341        let (carry, n7) = part1(carry + limbs[ 7]                                        + m(n3,l[4]) + m(n4,l[3]) + m(n5,l[2]) + m(n6,l[1]));
342        let (carry, n8) = part1(carry + limbs[ 8] + m(n0,l[8])                                        + m(n4,l[4]) + m(n5,l[3]) + m(n6,l[2]) + m(n7,l[1]));
343
344        // limbs is divisible by R now, so we can divide by R by simply storing the upper half as the result
345        let (carry, r0) = part2(carry + limbs[ 9]              + m(n1,l[8])                                        + m(n5,l[4]) + m(n6,l[3]) + m(n7,l[2]) + m(n8,l[1]));
346        let (carry, r1) = part2(carry + limbs[10]                           + m(n2,l[8])                                        + m(n6,l[4]) + m(n7,l[3]) + m(n8,l[2]));
347        let (carry, r2) = part2(carry + limbs[11]                                        + m(n3,l[8])                                        + m(n7,l[4]) + m(n8,l[3]));
348        let (carry, r3) = part2(carry + limbs[12]                                                     + m(n4,l[8])                                        + m(n8,l[4]));
349        let (carry, r4) = part2(carry + limbs[13]                                                                  + m(n5,l[8])                                       );
350        let (carry, r5) = part2(carry + limbs[14]                                                                               + m(n6,l[8])                          );
351        let (carry, r6) = part2(carry + limbs[15]                                                                                            + m(n7,l[8])             );
352        let (carry, r7) = part2(carry + limbs[16]                                                                                                         + m(n8,l[8]));
353        let         r8 = carry as u32;
354
355        // result may be >= l, so attempt to subtract l
356        Scalar29::sub(&Scalar29([r0,r1,r2,r3,r4,r5,r6,r7,r8]), l)
357    }

392    /// Takes a Scalar29 out of Montgomery form, i.e. computes `a/R (mod l)`
393    #[allow(clippy::wrong_self_convention)]
394    pub fn from_montgomery(&self) -> Scalar29 {
395        let mut limbs = [0u64; 17];
396        for i in 0..9 {
397            limbs[i] = self[i] as u64;
398        }
