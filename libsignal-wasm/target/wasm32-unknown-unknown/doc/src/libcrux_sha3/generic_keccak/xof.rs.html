6/// The internal keccak state that can also buffer inputs to absorb.
7/// This is used in the general xof APIs.
8pub(crate) struct KeccakXofState<
9    const PARALLEL_LANES: usize,
10    const RATE: usize,
11    STATE: KeccakItem<PARALLEL_LANES>,
12> {
13    inner: KeccakState<PARALLEL_LANES, STATE>,
14
15    // Buffer inputs on absorb.
16    buf: [[u8; RATE]; PARALLEL_LANES],
17
18    // Buffered length.
19    buf_len: usize,
20
21    // Needs sponge.
22    sponge: bool,
23}
24
25#[hax_lib::attributes]
26impl<const PARALLEL_LANES: usize, const RATE: usize, STATE: KeccakItem<PARALLEL_LANES>>
27    KeccakXofState<PARALLEL_LANES, RATE, STATE>
28{
29    /// An all zero block
30    pub(crate) const fn zero_block() -> [u8; RATE] {
31        [0u8; RATE]
32    }
33
34    /// Generate a new keccak xof state.
35    pub(crate) fn new() -> Self {
36        Self {
37            inner: KeccakState::new(),
38            buf: [Self::zero_block(); PARALLEL_LANES],
39            buf_len: 0,
40            sponge: false,
41        }
42    }
43
44    /// Absorb
45    ///
46    /// This function takes any number of bytes to absorb and buffers if it's not enough.
47    /// The function assumes that all input slices in `inputs` have the same length.
48    ///
49    /// Only a multiple of `RATE` blocks are absorbed.
50    /// For the remaining bytes [`absorb_final`] needs to be called.
51    ///
52    /// This works best with relatively small `inputs`.
53    #[inline(always)]
54    pub(crate) fn absorb(&mut self, inputs: &[&[u8]; PARALLEL_LANES])
55    where
56        KeccakState<PARALLEL_LANES, STATE>: Absorb<PARALLEL_LANES>,
57    {
58        let input_remainder_len = self.absorb_full(inputs);
59
60        // ... buffer the rest if there's not enough input (left).
61        if input_remainder_len > 0 {
62            debug_assert!(
63                self.buf_len == 0  // We consumed everything (or it was empty all along).
64                 || self.buf_len + input_remainder_len <= RATE
65            );
66
67            let input_len = inputs[0].len();
68
69            #[allow(clippy::needless_range_loop)]
70            for i in 0..PARALLEL_LANES {
71                self.buf[i][self.buf_len..self.buf_len + input_remainder_len]
72                    .copy_from_slice(&inputs[i][input_len - input_remainder_len..]);
73            }
74            self.buf_len += input_remainder_len;
75        }
76    }
77
78    // Note: consciously not inlining this function to avoid using too much stack
79    pub(crate) fn absorb_full(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) -> usize
80    where
81        KeccakState<PARALLEL_LANES, STATE>: Absorb<PARALLEL_LANES>,
82    {
83        debug_assert!(PARALLEL_LANES > 0);
84        debug_assert!(self.buf_len < RATE);
85        #[cfg(all(debug_assertions, not(hax)))]
86        {
87            for block in inputs {
88                debug_assert!(block.len() == inputs[0].len());
89            }
90        }
91
92        // Check if there are buffered bytes to absorb first and consume them.
93        let input_consumed = self.fill_buffer(inputs);
94
95        if input_consumed > 0 {
96            let mut borrowed = [[0u8; RATE].as_slice(); PARALLEL_LANES];
97            // We have a full block in the local buffer now.
98            #[allow(clippy::needless_range_loop)]
99            for i in 0..PARALLEL_LANES {
100                borrowed[i] = &self.buf[i];
101            }
102
103            self.inner.load_block::<RATE>(&borrowed, 0);
104            self.inner.keccakf1600();
105
106            // "empty" the local buffer
107            self.buf_len = 0;
108        }
109
110        // We only need to consume the rest of the input.
111        let input_to_consume = inputs[0].len() - input_consumed;
112
113        // Consume the (rest of the) input ...
114        let num_blocks = input_to_consume / RATE;
115        let remainder = input_to_consume % RATE;
116        for i in 0..num_blocks {
117            // We only get in here if `input_len / RATE > 0`.
118            self.inner
119                .load_block::<RATE>(inputs, input_consumed + i * RATE);
120            self.inner.keccakf1600();
121        }
122
123        remainder
124    }
125
126    /// Consume the internal buffer and the required amount of the input to pad to
127    /// `RATE`.
128    ///
129    /// Returns the `consumed` bytes from `inputs` if there's enough buffered
130    /// content to consume, and `0` otherwise.
131    /// If `consumed > 0` is returned, `self.buf` contains a full block to be
132    /// loaded.
133    // Note: consciously not inlining this function to avoid using too much stack
134    pub(crate) fn fill_buffer(&mut self, inputs: &[&[u8]; PARALLEL_LANES]) -> usize {
135        let input_len = inputs[0].len();
136        let mut consumed = 0;
137        if self.buf_len > 0 {
138            // There's something buffered internally to consume.
139            if self.buf_len + input_len >= RATE {
140                // We have enough data when combining the internal buffer and
141                // the input.
142                consumed = RATE - self.buf_len;
143
144                #[allow(clippy::needless_range_loop)]
145                for i in 0..PARALLEL_LANES {
146                    self.buf[i][self.buf_len..].copy_from_slice(&inputs[i][..consumed]);
147                }
148                self.buf_len += consumed;
149            }
150        }
151        consumed
152    }
153
154    /// Absorb a final block.
155    ///
156    /// The `inputs` block may be empty. Everything in the `inputs` block beyond
157    /// `RATE` bytes is ignored.
158    #[inline(always)]
159    pub(crate) fn absorb_final<const DELIMITER: u8>(&mut self, inputs: &[&[u8]; PARALLEL_LANES])
160    where
161        KeccakState<PARALLEL_LANES, STATE>: Absorb<PARALLEL_LANES>,
162    {
163        self.absorb(inputs);
164
165        let mut borrowed = [[0u8; RATE].as_slice(); PARALLEL_LANES];
166
167        #[allow(clippy::needless_range_loop)]
168        for i in 0..PARALLEL_LANES {
169            borrowed[i] = &self.buf[i];
170        }
171
172        self.inner
173            .load_last::<RATE, DELIMITER>(&borrowed, 0, self.buf_len);
174        self.inner.keccakf1600();
175    }
176}

178/// Squeeze we only implement for N = 1 right now.
179/// This is because it's not needed for N > 1 right now, but also because hax
180/// can't handle the required mutability for it.
181impl<const RATE: usize, STATE: KeccakItem<1>> KeccakXofState<1, RATE, STATE> {
182    /// Squeeze `N` x `LEN` bytes. Only `N = 1` for now.
183    #[inline(always)]
184    pub(crate) fn squeeze(&mut self, out: &mut [u8])
185    where
186        KeccakState<1, STATE>: Squeeze1<STATE>,
187    {
188        if self.sponge {
189            // If we called `squeeze` before, call f1600 first.
190            // We do it this way around so that we don't call f1600 at the end
191            // when we don't need it.
192            self.inner.keccakf1600();
193        }
194
195        let out_len = out.len();
196
197        if out_len > 0 {
198            if out_len <= RATE {
199                self.inner.squeeze::<RATE>(out, 0, out_len);
200            } else {
201                // How many blocks do we need to squeeze out?
202                let blocks = out_len / RATE;
203
204                for i in 0..blocks {
205                    // Here we know that we always have full blocks to write out.
206                    self.inner.keccakf1600();
207                    self.inner.squeeze::<RATE>(out, i * RATE, RATE);
208                }
209
210                let remaining = out_len % RATE;
211                if remaining > 0 {
212                    // Squeeze out the last partial block
213                    self.inner.keccakf1600();
214                    self.inner.squeeze::<RATE>(out, blocks * RATE, remaining);
215                }
216            }
217            self.sponge = true;
218        }
219    }
