3const SHA3_224_LEN: usize = 28;
4const SHA3_256_LEN: usize = 32;
5const SHA3_384_LEN: usize = 48;

8macro_rules! impl_hash_traits {
9    ($type:ident, $hasher:ident, $len:expr, $method:expr) => {
10        #[doc = concat!("A struct that implements [`libcrux_traits::digest`] traits.")]
11        #[doc = concat!("\n\n")]
12        #[doc = concat!("[`",stringify!($hasher), "`] is a convenient hasher for this struct.")]
13        pub struct $type;
14
15        #[doc = concat!("A hasher for [`",stringify!($type), "`].")]
16        pub type $hasher = libcrux_traits::digest::Hasher<$len, $type>;
17
18        impl libcrux_traits::digest::arrayref::Hash<$len> for $type {
19            #[inline(always)]
20            fn hash(
21                digest: &mut [u8; $len],
22                payload: &[u8],
23            ) -> Result<(), libcrux_traits::digest::arrayref::HashError> {
24                if payload.len() > u32::MAX as usize {
25                    return Err(libcrux_traits::digest::arrayref::HashError::InvalidPayloadLength);
26                }
27
28                $method(digest, payload);
29
30                Ok(())
31            }
32        }
33    };
34}
