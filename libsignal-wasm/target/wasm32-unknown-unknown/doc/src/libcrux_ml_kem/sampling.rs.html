6/// If `bytes` contains a set of uniformly random bytes, this function
7/// uniformly samples a ring element `√¢` that is treated as being the NTT representation
8/// of the corresponding polynomial `a`.
9///
10/// Since rejection sampling is used, it is possible the supplied bytes are
11/// not enough to sample the element, in which case an `Err` is returned and the
12/// caller must try again with a fresh set of bytes.
13///
14/// This function <strong>partially</strong> implements <strong>Algorithm 6</strong> of the NIST FIPS 203 standard,
15/// We say "partially" because this implementation only accepts a finite set of
16/// bytes as input and returns an error if the set is not enough; Algorithm 6 of
17/// the FIPS 203 standard on the other hand samples from an infinite stream of bytes
18/// until the ring element is filled. Algorithm 6 is reproduced below:
19///
20/// ```plaintext
21/// Input: byte stream B ‚àà ùîπ*.
22/// Output: array √¢ ‚àà ‚Ñ§‚ÇÇ‚ÇÖ‚ÇÜ.
23///
24/// i ‚Üê 0
25/// j ‚Üê 0
26/// while j < 256 do
27///     d‚ÇÅ ‚Üê B[i] + 256¬∑(B[i+1] mod 16)
28///     d‚ÇÇ ‚Üê ‚åäB[i+1]/16‚åã + 16¬∑B[i+2]
29///     if d‚ÇÅ < q then
30///         √¢[j] ‚Üê d‚ÇÅ
31///         j ‚Üê j + 1
32///     end if
33///     if d‚ÇÇ < q and j < 256 then
34///         √¢[j] ‚Üê d‚ÇÇ
35///         j ‚Üê j + 1
36///     end if
37///     i ‚Üê i + 3
38/// end while
39/// return √¢
40/// ```
41///
42/// The NIST FIPS 203 standard can be found at
43/// <https://csrc.nist.gov/pubs/fips/203/ipd>.
44#[inline(always)]
45fn sample_from_uniform_distribution_next<Vector: Operations, const K: usize, const N: usize>(
46    randomness: &[[u8; N]; K],
47    sampled_coefficients: &mut [usize; K],
48    out: &mut [[i16; 272]; K],
49) -> bool {
50    // Would be great to trigger auto-vectorization or at least loop unrolling here
51    for i in 0..K {
52        for r in 0..N / 24 {
53            if sampled_coefficients[i] < COEFFICIENTS_IN_RING_ELEMENT {
54                let sampled = Vector::rej_sample(
55                    &randomness[i][r * 24..(r * 24) + 24],
56                    &mut out[i][sampled_coefficients[i]..sampled_coefficients[i] + 16],
57                );
58                sampled_coefficients[i] += sampled;
59            }
60        }
61    }
62    let mut done = true;
63    for i in 0..K {
64        if sampled_coefficients[i] >= COEFFICIENTS_IN_RING_ELEMENT {
65            sampled_coefficients[i] = COEFFICIENTS_IN_RING_ELEMENT;
66        } else {
67            done = false
68        }
69    }
70    done
71}

75pub(super) fn sample_from_xof<const K: usize, Vector: Operations, Hasher: Hash<K>>(
76    seeds: &[[u8; 34]; K],
77) -> [PolynomialRingElement<Vector>; K] {
78    let mut sampled_coefficients: [usize; K] = [0; K];
79    let mut out: [[i16; 272]; K] = [[0; 272]; K];
80
81    let mut xof_state = Hasher::shake128_init_absorb_final(seeds);
82    let randomness = xof_state.shake128_squeeze_first_three_blocks();
83
84    let mut done = sample_from_uniform_distribution_next::<Vector, K, THREE_BLOCKS>(
85        &randomness,
86        &mut sampled_coefficients,
87        &mut out,
88    );
89
90    // Requiring more than 5 blocks to sample a ring element should be very
91    // unlikely according to:
92    // https://eprint.iacr.org/2023/708.pdf
93    // To avoid failing here, we squeeze more blocks out of the state until
94    // we have enough.
95    while !done {
96        let randomness = xof_state.shake128_squeeze_next_block();
97        done = sample_from_uniform_distribution_next::<Vector, K, BLOCK_SIZE>(
98            &randomness,
99            &mut sampled_coefficients,
100            &mut out,
101        );
102    }
103
104    out.map(|s| PolynomialRingElement::<Vector>::from_i16_array(&s[0..256]))
105}

107/// Given a series of uniformly random bytes in `randomness`, for some number `eta`,
108/// the `sample_from_binomial_distribution_{eta}` functions sample
109/// a ring element from a binomial distribution centered at 0 that uses two sets
110/// of `eta` coin flips. If, for example,
111/// `eta = ETA`, each ring coefficient is a value `v` such
112/// such that `v ‚àà {-ETA, -ETA + 1, ..., 0, ..., ETA + 1, ETA}` and:
113///
114/// ```plaintext
115/// - If v < 0, Pr[v] = Pr[-v]
116/// - If v >= 0, Pr[v] = BINOMIAL_COEFFICIENT(2 * ETA; ETA - v) / 2 ^ (2 * ETA)
117/// ```
118///
119/// The values `v < 0` are mapped to the appropriate `KyberFieldElement`.
120///
121/// The expected value is:
122///
123/// ```plaintext
124/// E[X] = (-ETA)Pr[-ETA] + (-(ETA - 1))Pr[-(ETA - 1)] + ... + (ETA - 1)Pr[ETA - 1] + (ETA)Pr[ETA]
125///      = 0 since Pr[-v] = Pr[v] when v < 0.
126/// ```
127///
128/// And the variance is:
129///
130/// ```plaintext
131/// Var(X) = E[(X - E[X])^2]
132///        = E[X^2]
133///        = sum_(v=-ETA to ETA)v^2 * (BINOMIAL_COEFFICIENT(2 * ETA; ETA - v) / 2^(2 * ETA))
134///        = ETA / 2
135/// ```
136///
137/// This function implements <strong>Algorithm 7</strong> of the NIST FIPS 203 standard, which is
138/// reproduced below:
139///
140/// ```plaintext
141/// Input: byte array B ‚àà ùîπ^{64Œ∑}.
142/// Output: array f ‚àà ‚Ñ§‚ÇÇ‚ÇÖ‚ÇÜ.
143///
144/// b ‚Üê BytesToBits(B)
145/// for (i ‚Üê 0; i < 256; i++)
146///     x ‚Üê ‚àë(j=0 to Œ∑ - 1) b[2iŒ∑ + j]
147///     y ‚Üê ‚àë(j=0 to Œ∑ - 1) b[2iŒ∑ + Œ∑ + j]
148///     f[i] ‚Üê x‚àíy mod q
149/// end for
150/// return f
151/// ```
152///
153/// The NIST FIPS 203 standard can be found at
154/// <https://csrc.nist.gov/pubs/fips/203/ipd>.
155#[hax_lib::requires(randomness.len() == 2 * 64)]

160// ))))]

162#[hax_lib::fstar::options("--z3rlimit 800")]
163fn sample_from_binomial_distribution_2<Vector: Operations>(
164    randomness: &[u8],
165) -> PolynomialRingElement<Vector> {
166    hax_lib::fstar!(
167        "assert (v (sz 2 *! sz 64) == 128);
168        assert (Seq.length $randomness == 128)"
169    );
170    let mut sampled_i16s = [0i16; 256];
171
172    cloop! {
173        for (chunk_number, byte_chunk) in randomness.chunks_exact(4).enumerate() {
174            let random_bits_as_u32: u32 = (byte_chunk[0] as u32)
175                | (byte_chunk[1] as u32) << 8
176                | (byte_chunk[2] as u32) << 16
177                | (byte_chunk[3] as u32) << 24;
178
179            let even_bits = random_bits_as_u32 & 0x55555555;
180            let odd_bits = (random_bits_as_u32 >> 1) & 0x55555555;
181            hax_lib::fstar!(r#"logand_lemma $random_bits_as_u32 (mk_u32 1431655765);
182                logand_lemma ($random_bits_as_u32 >>! (mk_i32 1)) (mk_u32 1431655765)"#);
183            let coin_toss_outcomes = even_bits + odd_bits;
184
185            cloop! {
186                for outcome_set in (0..32u32).step_by(4) { // u32::BITS
187                    let outcome_1 = ((coin_toss_outcomes >> outcome_set) & 0x3) as i16;
188                    let outcome_2 = ((coin_toss_outcomes >> (outcome_set + 2)) & 0x3) as i16;
189                    hax_lib::fstar!(r#"logand_lemma ($coin_toss_outcomes >>! $outcome_set <: u32) (mk_u32 3);
190                        logand_lemma ($coin_toss_outcomes >>! ($outcome_set +! (mk_u32 2) <: u32) <: u32) (mk_u32 3);
191                        assert (v $outcome_1 >= 0 /\ v $outcome_1 <= 3);
192                        assert (v $outcome_2 >= 0 /\ v $outcome_2 <= 3);
193                        assert (v $chunk_number <= 31);
194                        assert (v (sz 8 *! $chunk_number <: usize) <= 248);
195                        assert (v (cast ($outcome_set >>! (mk_i32 2) <: u32) <: usize) <= 7)"#);
196
197                    let offset = (outcome_set >> 2) as usize;
198                    sampled_i16s[8 * chunk_number + offset] = outcome_1 - outcome_2;
199                }
200            }
201        }
202    }
203    PolynomialRingElement::from_i16_array(&sampled_i16s)
204}

213#[hax_lib::fstar::options("--z3rlimit 800")]
214fn sample_from_binomial_distribution_3<Vector: Operations>(
215    randomness: &[u8],
216) -> PolynomialRingElement<Vector> {
217    hax_lib::fstar!(
218        "assert (v (sz 3 *! sz 64) == 192);
219        assert (Seq.length $randomness == 192)"
220    );
221    let mut sampled_i16s = [0i16; 256];
222
223    cloop! {
224        for (chunk_number, byte_chunk) in randomness.chunks_exact(3).enumerate() {
225            let random_bits_as_u24: u32 =
226                (byte_chunk[0] as u32) | (byte_chunk[1] as u32) << 8 | (byte_chunk[2] as u32) << 16;
227
228            let first_bits = random_bits_as_u24 & 0x00249249;
229            let second_bits = (random_bits_as_u24 >> 1) & 0x00249249;
230            let third_bits = (random_bits_as_u24 >> 2) & 0x00249249;
231            hax_lib::fstar!(r#"logand_lemma $random_bits_as_u24 (mk_u32 2396745);
232                logand_lemma ($random_bits_as_u24 >>! (mk_i32 1) <: u32) (mk_u32 2396745);
233                logand_lemma ($random_bits_as_u24 >>! (mk_i32 2) <: u32) (mk_u32 2396745)"#);
234
235            let coin_toss_outcomes = first_bits + second_bits + third_bits;
236
237            cloop! {
238                for outcome_set in (0..24).step_by(6) {
239                    let outcome_1 = ((coin_toss_outcomes >> outcome_set) & 0x7) as i16;
240                    let outcome_2 = ((coin_toss_outcomes >> (outcome_set + 3)) & 0x7) as i16;
241                    hax_lib::fstar!(r#"logand_lemma ($coin_toss_outcomes >>! $outcome_set <: u32) (mk_u32 7);
242                        logand_lemma ($coin_toss_outcomes >>! ($outcome_set +! (mk_i32 3) <: i32) <: u32) (mk_u32 7);
243                        assert (v $outcome_1 >= 0 /\ v $outcome_1 <= 7);
244                        assert (v $outcome_2 >= 0 /\ v $outcome_2 <= 7);
245                        assert (v $chunk_number <= 63);
246                        assert (v (sz 4 *! $chunk_number <: usize) <= 252);
247                        assert (v (cast ($outcome_set /! (mk_i32 6) <: i32) <: usize) <= 3)"#);
248
249                    let offset = (outcome_set / 6) as usize;
250                    sampled_i16s[4 * chunk_number + offset] = outcome_1 - outcome_2;
251                }
252            }
253        }
254    }
255    PolynomialRingElement::from_i16_array(&sampled_i16s)
256}

266pub(super) fn sample_from_binomial_distribution<const ETA: usize, Vector: Operations>(
267    randomness: &[u8],
268) -> PolynomialRingElement<Vector> {
269    hax_lib::fstar!(
270        r#"assert (
271        (v (cast $ETA <: u32) == 2) \/
272        (v (cast $ETA <: u32) == 3))"#
273    );
274    match ETA as u32 {
275        2 => sample_from_binomial_distribution_2(randomness),
276        3 => sample_from_binomial_distribution_3(randomness),
277        _ => unreachable!(),
278    }
