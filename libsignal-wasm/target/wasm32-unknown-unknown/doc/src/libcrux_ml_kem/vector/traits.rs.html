8// We define a trait that allows us to talk about the contents of a vector.
9// This is used extensively in pre- and post-conditions to reason about the code.
10// The trait is duplicated for Eurydice to avoid the trait inheritance between Operations and Repr
11// This is needed because of this issue: https://github.com/AeneasVerif/eurydice/issues/111
12#[cfg(hax)]
13#[hax_lib::attributes]
14pub trait Repr: Copy + Clone {
15    #[requires(true)]
16    fn repr(&self) -> [i16; 16];
17}
18
19#[cfg(any(eurydice, not(hax)))]
20pub trait Repr {}

22#[cfg(hax)]
23mod spec {
24    pub(crate) fn add_pre(lhs: &[i16; 16], rhs: &[i16; 16]) -> hax_lib::Prop {
25        hax_lib::fstar_prop_expr!(
26            r#"forall i. i < 16 ==> 
27            Spec.Utils.is_intb (pow2 15 - 1) 
28                (v (Seq.index ${lhs} i) + v (Seq.index ${rhs} i))"#
29        )
30    }
31
32    pub(crate) fn add_post(lhs: &[i16; 16], rhs: &[i16; 16], result: &[i16; 16]) -> hax_lib::Prop {
33        hax_lib::fstar_prop_expr!(
34            r#"forall i. i < 16 ==> 
35            (v (Seq.index ${result} i) == 
36            v (Seq.index ${lhs} i) + v (Seq.index ${rhs} i))"#
37        )
38    }
39}
