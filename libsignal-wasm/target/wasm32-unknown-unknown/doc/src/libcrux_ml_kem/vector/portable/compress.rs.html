7/// The `compress_*` functions implement the `Compress` function specified in the NIST FIPS
8/// 203 standard (Page 18, Expression 4.5), which is defined as:
9///
10/// ```plaintext
11/// Compress_d: ℤq -> ℤ_{2ᵈ}
12/// Compress_d(x) = ⌈(2ᵈ/q)·x⌋
13/// ```
14///
15/// Since `⌈x⌋ = ⌊x + 1/2⌋` we have:
16///
17/// ```plaintext
18/// Compress_d(x) = ⌊(2ᵈ/q)·x + 1/2⌋
19///               = ⌊(2^{d+1}·x + q) / 2q⌋
20/// ```
21///
22/// For further information about the function implementations, consult the
23/// `implementation_notes.pdf` document in this directory.
24///
25/// The NIST FIPS 203 standard can be found at
26/// <https://csrc.nist.gov/pubs/fips/203/ipd>.
27#[hax_lib::fstar::options("--z3rlimit 200 --ext context_pruning")]
28#[hax_lib::fstar::before(r#"[@ "opaque_to_smt"]"#)]
29#[cfg_attr(hax, hax_lib::requires(fe < (FIELD_MODULUS as u16)))]
30#[cfg_attr(hax, hax_lib::ensures(|result| fstar!(r#"((833 <= v fe && v fe <= 2496) ==> v result == 1) /\
31						    (~(833 <=  v fe && v fe <= 2496) ==> v result == 0)"#)))]
32pub(crate) fn compress_message_coefficient(fe: U16) -> U8 {
33    // The approach used here is inspired by:
34    // https://github.com/cloudflare/circl/blob/main/pke/kyber/internal/common/poly.go#L150
35
36    // If 833 <= fe <= 2496,
37    // then -832 <= shifted <= 831
38    let shifted: I16 = 1664.classify() - (fe.as_i16());
39    hax_lib::fstar!(r#"assert (v $shifted == 1664 - v $fe)"#);
40
41    // If shifted < 0, then
42    // (shifted >> 15) ^ shifted = flip_bits(shifted) = -shifted - 1, and so
43    // if -832 <= shifted < 0 then 0 < shifted_positive <= 831
44    //
45    // If shifted >= 0 then
46    // (shifted >> 15) ^ shifted = shifted, and so
47    // if 0 <= shifted <= 831 then 0 <= shifted_positive <= 831
48    let mask = shifted >> 15;
49
50    hax_lib::fstar!(
51        "assert (v $mask = v $shifted / pow2 15);
52        assert (if v $shifted < 0 then $mask = ones else $mask = zero)"
53    );
54
55    let shifted_to_positive: I16 = mask ^ shifted;
56
57    hax_lib::fstar!(
58        "logxor_lemma $shifted $mask;
59        assert (v $shifted < 0 ==> v $shifted_to_positive = v (lognot $shifted));
60        neg_equiv_lemma $shifted;
61        assert (v (lognot $shifted) = -(v $shifted) -1);
62        assert (v $shifted >= 0 ==> v $shifted_to_positive = v ($mask `logxor` $shifted));
63        assert (v $shifted >= 0 ==> $mask = zero);
64        assert (v $shifted >= 0 ==> $mask ^. $shifted = $shifted);
65        assert (v $shifted >= 0 ==> v $shifted_to_positive = v $shifted);
66        assert ($shifted_to_positive >=. mk_i16 0)"
67    );
68
69    let shifted_positive_in_range: I16 = shifted_to_positive - 832;
70
71    hax_lib::fstar!(
72        "assert (1664 - v $fe >= 0 ==> v $shifted_positive_in_range == 832 - v $fe);
73        assert (1664 - v $fe < 0 ==> v $shifted_positive_in_range == -2497 + v $fe)"
74    );
75
76    // If x <= 831, then x - 832 <= -1, and so x - 832 < 0, which means
77    // the most significant bit of shifted_positive_in_range will be 1.
78    let r0: I16 = shifted_positive_in_range >> 15;
79    let r1: I16 = r0 & 1i16;
80    let res = r1.as_u8();
81
82    hax_lib::fstar!(
83        r#"assert (v $r0 = v $shifted_positive_in_range / pow2 15);
84        assert (if v $shifted_positive_in_range < 0 then $r0 = ones else $r0 = zero);
85        logand_lemma (mk_i16 1) $r0; 
86        assert (if v $shifted_positive_in_range < 0 then $r1 = mk_i16 1 else $r1 = mk_i16 0);
87        assert ((v $fe >= 833 && v $fe <= 2496) ==> $r1 = mk_i16 1);
88        assert (v $fe < 833 ==> $r1 = mk_i16 0);
89        assert (v $fe > 2496 ==> $r1 = mk_i16 0);
90        assert (v $res = v $r1)"#
91    );
92
93    res
94}

108pub(crate) fn compress_ciphertext_coefficient(coefficient_bits: u8, fe: U16) -> FieldElement {
109    // hax_debug_assert!(
110    //     coefficient_bits == 4
111    //         || coefficient_bits == 5
112    //         || coefficient_bits == 10
113    //         || coefficient_bits == 11
114    // );
115    // hax_debug_assert!(fe <= (FIELD_MODULUS as u16));
116
117    // This has to be constant time due to:
118    // https://groups.google.com/a/list.nist.gov/g/pqc-forum/c/ldX0ThYJuBo/m/ovODsdY7AwAJ
119    let mut compressed = (fe.as_u64()) << coefficient_bits;
120    compressed += 1664 as u64;
121
122    compressed *= 10_321_340;
123    compressed >>= 35;
124
125    get_n_least_significant_bits(coefficient_bits, compressed.as_u32()).as_i16()
126}

148pub(crate) fn compress_1(mut a: PortableVector) -> PortableVector {
149    hax_lib::fstar!(
150        "assert (forall (i:nat). i < 16 ==> (cast (${a}.f_elements.[ sz i ]) <: u16) <.
151        (cast ($FIELD_MODULUS) <: u16))"
152    );
153
154    for i in 0..FIELD_ELEMENTS_IN_VECTOR {
155        hax_lib::loop_invariant!(|i: usize| {
156            fstar!(
157                r#"(v $i < 16 ==> (forall (j:nat). (j >= v $i /\ j < 16) ==>
158            v (cast (${a}.f_elements.[ sz j ]) <: u16) < v (cast ($FIELD_MODULUS) <: u16))) /\
159            (forall (j:nat). j < v $i ==> v (${a}.f_elements.[ sz j ] <: i16) >= 0 /\
160                v (${a}.f_elements.[ sz j ] <: i16) < 2)"#
161            )
162        });
163
164        hax_lib::fstar!(
165            "compress_message_coefficient_range_helper (cast (${a}.f_elements.[ $i ]) <: u16)"
166        );
167
168        a.elements[i] = compress_message_coefficient(a.elements[i].as_u16()).as_i16();
169
170        hax_lib::fstar!(
171            r#"assert (v (${a}.f_elements.[ $i ] <: i16) >= 0 /\
172            v (${a}.f_elements.[ $i ] <: i16) < 2)"#
173        );
174    }
175
176    hax_lib::fstar!(
177        r#"assert (forall (i:nat). i < 16 ==> v (${a}.f_elements.[ sz i ] <: i16) >= 0 /\
178        v (${a}.f_elements.[ sz i ] <: i16) < 2)"#
179    );
180
181    a
182}

196pub(crate) fn compress<const COEFFICIENT_BITS: i32>(mut a: PortableVector) -> PortableVector {
197    hax_lib::fstar!(
198        "assert (v (cast ($COEFFICIENT_BITS) <: u8) == v $COEFFICIENT_BITS);
199        assert (v (cast ($COEFFICIENT_BITS) <: u32) == v $COEFFICIENT_BITS);
200        assert (v (cast ($FIELD_MODULUS) <: u16) == 3329)"
201    );
202    hax_lib::fstar!(
203        "assert (forall (i:nat). i < 16 ==> 
204            (cast (${a}.f_elements.[ sz i ]) <: u16) <.
205            (cast ($FIELD_MODULUS) <: u16))"
206    );
207
208    for i in 0..FIELD_ELEMENTS_IN_VECTOR {
209        hax_lib::loop_invariant!(|i: usize| {
210            fstar!(
211                r#"((forall (j:nat). (j >= v $i /\ j < 16) ==>
212                      v (cast (${a}.f_elements.[ sz j ]) <: u16) < v (cast ($FIELD_MODULUS) <: u16))) /\
213                    (forall (j:nat). j < v $i ==> v (${a}.f_elements.[ sz j ] <: i16) >= 0 /\
214                     v (${a}.f_elements.[ sz j ] <: i16) < pow2 (v $COEFFICIENT_BITS)))"#
215            )
216        });
217
218        a.elements[i] =
219            compress_ciphertext_coefficient(COEFFICIENT_BITS as u8, a.elements[i].as_u16())
220                .as_i16();
221
222        hax_lib::fstar!(
223            r#"assert (v (${a}.f_elements.[ $i ] <: i16) >= 0 /\
224            v (${a}.f_elements.[ $i ] <: i16) < pow2 (v $COEFFICIENT_BITS))"#
225        );
226    }
227    hax_lib::fstar!(
228        r#"assert (forall (i:nat). i < 16 ==> 
229                    (v (${a}.f_elements.[ sz i ] <: i16) >= 0 /\
230                     v (${a}.f_elements.[ sz i ] <: i16) < pow2 (v $COEFFICIENT_BITS)))"#
231    );
232
233    a
234}

243pub(crate) fn decompress_1(a: PortableVector) -> PortableVector {
244    let z = zero();
245
246    hax_lib::fstar!("assert(forall i. Seq.index ${z}.f_elements i == mk_i16 0)");
247    hax_lib::fstar!(
248        r#"assert(forall i. let x = Seq.index ${a}.f_elements i in 
249                                      ((0 - v x) == 0 \/ (0 - v x) == -1))"#
250    );
251    hax_lib::fstar!(
252        r#"assert(forall i. i < 16 ==>
253                                      Spec.Utils.is_intb (pow2 15 - 1) 
254                                        (0 - v (Seq.index ${a}.f_elements i)))"#
255    );
256
257    let s = sub(z, &a);
258
259    hax_lib::fstar!(
260        r#"assert(forall i. Seq.index ${s}.f_elements i == mk_i16 0 \/ 
261                                      Seq.index ${s}.f_elements i == mk_i16 (-1))"#
262    );
263
264    let res = bitwise_and_with_constant(s, 1665);
265
266    hax_lib::fstar!(
267        r#"assert(forall i. Seq.index ${res}.f_elements i == mk_i16 0 \/ 
268                                      Seq.index ${res}.f_elements i == mk_i16 1665)"#
269    );
270
271    res
272}

285pub(crate) fn decompress_ciphertext_coefficient<const COEFFICIENT_BITS: i32>(
286    mut a: PortableVector,
287) -> PortableVector {
288    hax_lib::fstar!(
289        "assert_norm (pow2 1 == 2);
290        assert_norm (pow2 4 == 16);
291        assert_norm (pow2 5 == 32);
292        assert_norm (pow2 10 == 1024);
293        assert_norm (pow2 11 == 2048)"
294    );
295
296    for i in 0..FIELD_ELEMENTS_IN_VECTOR {
297        hax_lib::loop_invariant!(|i: usize| {
298            fstar!(
299                r#"(v $i < 16 ==> (forall (j:nat). (j >= v $i /\ j < 16) ==>
300            v (Seq.index ${a}.f_elements j) >= 0 /\  v (Seq.index ${a}.f_elements j) < pow2 (v $COEFFICIENT_BITS))) /\
301            (forall (j:nat). j < v $i ==>
302                (v (Seq.index ${a}.f_elements j) >= 0 /\ v (Seq.index ${a}.f_elements j) < v $FIELD_MODULUS))"#
303            )
304        });
305        hax_lib::fstar!(
306            "assert (v (${a}.f_elements.[ $i ] <: i16) < pow2 11);
307          assert (v (${a}.f_elements.[ $i ] <: i16) ==
308            v (cast (${a}.f_elements.[ $i ] <: i16) <: i32));
309          assert (v ($FIELD_MODULUS <: i16) ==
310            v (cast ($FIELD_MODULUS <: i16) <: i32));
311          assert (v ((cast (${a}.f_elements.[ $i ] <: i16) <: i32) *!
312            (cast ($FIELD_MODULUS <: i16) <: i32)) ==
313              v (cast (${a}.f_elements.[ $i ] <: i16) <: i32) *
314                v (cast ($FIELD_MODULUS <: i16) <: i32))"
315        );
316
317        let mut decompressed = a.elements[i].as_i32() * FIELD_MODULUS.classify().as_i32();
318
319        hax_lib::fstar!(
320            "assert (v ($decompressed <<! mk_i32 1) == v $decompressed * 2);
321          assert (v (mk_i32 1 <<! $COEFFICIENT_BITS) == pow2 (v $COEFFICIENT_BITS));
322          assert (v (($decompressed <<! mk_i32 1) +! (mk_i32 1 <<! $COEFFICIENT_BITS)) ==
323            v ($decompressed <<! mk_i32 1) + v (mk_i32 1 <<! $COEFFICIENT_BITS))"
324        );
325
326        decompressed = (decompressed << 1) + (1i32 << COEFFICIENT_BITS);
327
328        hax_lib::fstar!(
329            "assert (v ($COEFFICIENT_BITS +! mk_i32 1) == v $COEFFICIENT_BITS + 1);
330          assert (v ($decompressed >>! ($COEFFICIENT_BITS +! mk_i32 1 <: i32)) ==
331            v $decompressed / pow2 (v $COEFFICIENT_BITS + 1))"
332        );
333
334        decompressed = decompressed >> (COEFFICIENT_BITS + 1);
335
336        hax_lib::fstar!(
337            "assert (v $decompressed < v $FIELD_MODULUS);
338          assert (v (cast $decompressed <: i16) < v $FIELD_MODULUS)"
339        );
340
341        a.elements[i] = decompressed.as_i16();
342    }
343
344    a
