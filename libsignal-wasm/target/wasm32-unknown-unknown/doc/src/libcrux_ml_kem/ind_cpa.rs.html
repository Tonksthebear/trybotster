139pub(crate) fn serialize_vector<const K: usize, Vector: Operations>(
140    key: &[PolynomialRingElement<Vector>; K],
141    out: &mut [u8],
142) {
143    hax_lib::fstar!(r#"assert_norm (Spec.MLKEM.polynomial_d 12 == Spec.MLKEM.polynomial)"#);
144
145    cloop! {
146        for (i, re) in key.into_iter().enumerate() {
147            hax_lib::loop_invariant!(|i: usize| {
148                fstar!(r#"${out.len()} == Spec.MLKEM.v_CPA_PUBLIC_KEY_SIZE $K /\
149                    (v $i < v $K ==>
150                    Libcrux_ml_kem.Polynomial.is_bounded_poly 3328 (Seq.index $key (v $i))) /\
151                    (forall (j: nat). j < v $i ==>
152                    (j + 1) * v $BYTES_PER_RING_ELEMENT <= Seq.length $out /\
153                    (Seq.slice $out (j * v $BYTES_PER_RING_ELEMENT) ((j + 1) * v $BYTES_PER_RING_ELEMENT) ==
154                        Spec.MLKEM.byte_encode 12 (Libcrux_ml_kem.Polynomial.to_spec_poly_t #$:Vector (Seq.index $key j))))"#
155                )
156            });
157
158            out[i * BYTES_PER_RING_ELEMENT..(i + 1) * BYTES_PER_RING_ELEMENT]
159            .copy_from_slice(&serialize_uncompressed_ring_element(&re));
160
161            hax_lib::fstar!(r#"
162                let lemma_aux (j: nat{ j < v $i }) : Lemma
163                (Seq.slice out (j * v $BYTES_PER_RING_ELEMENT) ((j + 1) * v $BYTES_PER_RING_ELEMENT) ==
164                    Spec.MLKEM.byte_encode 12 (Libcrux_ml_kem.Polynomial.to_spec_poly_t #$:Vector (Seq.index $key j))) =
165                eq_intro (Seq.slice out (j * v $BYTES_PER_RING_ELEMENT) ((j + 1) * v $BYTES_PER_RING_ELEMENT))
166                (Spec.MLKEM.byte_encode 12 (Libcrux_ml_kem.Polynomial.to_spec_poly_t #$:Vector (Seq.index $key j)))
167                in
168                Classical.forall_intro lemma_aux"#

170        }
171    }
172
173    hax_lib::fstar!(
174        r#"assert (Spec.MLKEM.coerce_vector_12 (Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector $key) ==

176        reveal_opaque (`%Spec.MLKEM.vector_encode_12) (Spec.MLKEM.vector_encode_12 #$K);
177        eq_intro $out
178          (Spec.MLKEM.vector_encode_12 #$K
179            (Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector $key))"#
180    );

414/// This function implements most of <strong>Algorithm 12</strong> of the
415/// NIST FIPS 203 specification; this is the Kyber CPA-PKE key generation algorithm.
416///
417/// We say "most of" since Algorithm 12 samples the required randomness within
418/// the function itself, whereas this implementation expects it to be provided
419/// through the `key_generation_seed` parameter.
420///
421/// Algorithm 12 is reproduced below:
422///
423/// ```plaintext
424/// Output: encryption key ek‚Çö‚Çñ‚Çë ‚àà ùîπ^{384k+32}.
425/// Output: decryption key dk‚Çö‚Çñ‚Çë ‚àà ùîπ^{384k}.
426///
427/// d ‚Üê$ B
428/// (œÅ,œÉ) ‚Üê G(d)
429/// N ‚Üê 0
430/// for (i ‚Üê 0; i < k; i++)
431///     for(j ‚Üê 0; j < k; j++)
432///         √Ç[i,j] ‚Üê SampleNTT(XOF(œÅ, i, j))
433///     end for
434/// end for
435/// for(i ‚Üê 0; i < k; i++)
436///     s[i] ‚Üê SamplePolyCBD_{Œ∑‚ÇÅ}(PRF_{Œ∑‚ÇÅ}(œÉ,N))
437///     N ‚Üê N + 1
438/// end for
439/// for(i ‚Üê 0; i < k; i++)

480>(
481    key_generation_seed: &[u8],
482    private_key: &mut IndCpaPrivateKeyUnpacked<K, Vector>,
483    public_key: &mut IndCpaPublicKeyUnpacked<K, Vector>,
484) {
485    // (œÅ,œÉ) := G(d) for Kyber, (œÅ,œÉ) := G(d || K) for ML-KEM
486    let hashed = Scheme::cpa_keygen_seed::<K, Hasher>(key_generation_seed);
487    let (seed_for_A, seed_for_secret_and_error) = hashed.split_at(32);
488
489    hax_lib::fstar!(

494
495    hax_lib::fstar!(

498    );

500    hax_lib::fstar!("eq_intro $seed_for_secret_and_error (Seq.slice $prf_input 0 32)");

503            &mut private_key.secret_as_ntt,
504            &prf_input,
505            0,
506        );
507

510        &mut error_as_ntt,
511        &prf_input,
512        domain_separator,
513    );
514
515    // tÀÜ := AÀÜ ‚ó¶ sÀÜ + eÀÜ
516    compute_As_plus_e(
517        &mut public_key.t_as_ntt,
518        &public_key.A,
519        &private_key.secret_as_ntt,
520        &error_as_ntt,
521    );
522
523    public_key.seed_for_A = seed_for_A.try_into().unwrap();
524
525    hax_lib::fstar!(
526        r#"let (((t_as_ntt,seed_for_A), matrix_A_as_ntt), secret_as_ntt), valid =
527        Spec.MLKEM.ind_cpa_generate_keypair_unpacked $K $key_generation_seed in
528        assert (valid ==>
529            ((Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector ${public_key.t_as_ntt}) ==
530              t_as_ntt) /\ (${public_key.seed_for_A} == seed_for_A) /\
531            (Libcrux_ml_kem.Polynomial.to_spec_matrix_t #$K #$:Vector ${public_key.A} == matrix_A_as_ntt) /\
532            ((Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector ${private_key.secret_as_ntt}) ==
533              secret_as_ntt));
534        assert ((forall (i: nat). i < v $K ==>
535              Libcrux_ml_kem.Polynomial.is_bounded_poly 3328 (Seq.index ${private_key.secret_as_ntt} i)) /\
536          (forall (i: nat). i < v $K ==>
537              Libcrux_ml_kem.Polynomial.is_bounded_poly 3328 (Seq.index ${public_key.t_as_ntt} i)))"#
538    );
539
540    // For encapsulation, we need to store A not AÀÜ, and so we untranspose A
541    // However, we pass A_transpose here and let the IND-CCA layer do the untranspose.
542    // We could do it here, but then we would pay the performance cost (if any) for the packed API as well.
543}

670/// This function implements <strong>Algorithm 13</strong> of the
671/// NIST FIPS 203 specification; this is the Kyber CPA-PKE encryption algorithm.
672///
673/// Algorithm 13 is reproduced below:
674///
675/// ```plaintext
676/// Input: encryption key ek‚Çö‚Çñ‚Çë ‚àà ùîπ^{384k+32}.
677/// Input: message m ‚àà ùîπ^{32}.
678/// Input: encryption randomness r ‚àà ùîπ^{32}.
679/// Output: ciphertext c ‚àà ùîπ^{32(d·µ§k + d·µ•)}.
680///
681/// N ‚Üê 0
682/// tÃÇ ‚Üê ByteDecode‚ÇÅ‚ÇÇ(ek‚Çö‚Çñ‚Çë[0:384k])
683/// œÅ ‚Üê ek‚Çö‚Çñ‚Çë[384k: 384k + 32]
684/// for (i ‚Üê 0; i < k; i++)
685///     for(j ‚Üê 0; j < k; j++)
686///         √Ç[i,j] ‚Üê SampleNTT(XOF(œÅ, i, j))
687///     end for
688/// end for
689/// for(i ‚Üê 0; i < k; i++)
690///     r[i] ‚Üê SamplePolyCBD_{Œ∑‚ÇÅ}(PRF_{Œ∑‚ÇÅ}(r,N))
691///     N ‚Üê N + 1
692/// end for
693/// for(i ‚Üê 0; i < k; i++)
694///     e‚ÇÅ[i] ‚Üê SamplePolyCBD_{Œ∑‚ÇÇ}(PRF_{Œ∑‚ÇÇ}(r,N))
695///     N ‚Üê N + 1
696/// end for
697/// e‚ÇÇ ‚Üê SamplePolyCBD_{Œ∑‚ÇÇ}(PRF_{Œ∑‚ÇÇ}(r,N))
698/// rÃÇ ‚Üê NTT(r)
699/// u ‚Üê NTT-¬π(√Ç·µÄ ‚ó¶ rÃÇ) + e‚ÇÅ
700/// Œº ‚Üê Decompress‚ÇÅ(ByteDecode‚ÇÅ(m)))
701/// v ‚Üê NTT-¬π(tÃÇ·µÄ ‚ó¶ rÀÜ) + e‚ÇÇ + Œº
702/// c‚ÇÅ ‚Üê ByteEncode_{d·µ§}(Compress_{d·µ§}(u))
703/// c‚ÇÇ ‚Üê ByteEncode_{d·µ•}(Compress_{d·µ•}(v))
704/// return c ‚Üê (c‚ÇÅ ‚Äñ c‚ÇÇ)
705/// ```
706///
707/// The NIST FIPS 203 standard can be found at
708/// <https://csrc.nist.gov/pubs/fips/203/ipd>.
709#[allow(non_snake_case)]
710#[hax_lib::fstar::options("--z3rlimit 800 --ext context_pruning")]
711#[hax_lib::requires(fstar!(r#"Spec.MLKEM.is_rank $K /\
712      $ETA1 == Spec.MLKEM.v_ETA1 $K /\
713      $ETA1_RANDOMNESS_SIZE == Spec.MLKEM.v_ETA1_RANDOMNESS_SIZE $K /\
714      $ETA2 == Spec.MLKEM.v_ETA2 $K /\
715      $ETA2_RANDOMNESS_SIZE == Spec.MLKEM.v_ETA2_RANDOMNESS_SIZE $K /\
716      $C1_LEN == Spec.MLKEM.v_C1_SIZE $K /\
717      $C2_LEN == Spec.MLKEM.v_C2_SIZE $K /\
718      $U_COMPRESSION_FACTOR == Spec.MLKEM.v_VECTOR_U_COMPRESSION_FACTOR $K /\
719      $V_COMPRESSION_FACTOR == Spec.MLKEM.v_VECTOR_V_COMPRESSION_FACTOR $K /\
720      $BLOCK_LEN == Spec.MLKEM.v_C1_BLOCK_SIZE $K /\
721      $CIPHERTEXT_SIZE == Spec.MLKEM.v_CPA_CIPHERTEXT_SIZE $K /\
722      length $randomness == Spec.MLKEM.v_SHARED_SECRET_SIZE"#))]
723#[hax_lib::ensures(|result|
724    fstar!(r#"$result == Spec.MLKEM.ind_cpa_encrypt_unpacked $K $message $randomness
725        (Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector ${public_key.t_as_ntt})

775#[inline(always)]
776pub(crate) fn encrypt_c1<
777    const K: usize,
778    const C1_LEN: usize,
779    const U_COMPRESSION_FACTOR: usize,
780    const BLOCK_LEN: usize,
781    const ETA1: usize,
782    const ETA1_RANDOMNESS_SIZE: usize,
783    const ETA2: usize,
784    const ETA2_RANDOMNESS_SIZE: usize,
785    Vector: Operations,
786    Hasher: Hash<K>,
787>(
788    randomness: &[u8],
789    matrix: &[[PolynomialRingElement<Vector>; K]; K],
790    ciphertext: &mut [u8], // C1_LEN
791) -> (
792    [PolynomialRingElement<Vector>; K],
793    PolynomialRingElement<Vector>,
794) {
795    // for i from 0 to k‚àí1 do
796    //     r[i] := CBD{Œ∑1}(PRF(r, N))
797    //     N := N + 1
798    // end for
799    // rÀÜ := NTT(r)
800    let mut prf_input: [u8; 33] = into_padded_array(randomness);
801
802    let mut r_as_ntt = from_fn(|_i| PolynomialRingElement::<Vector>::ZERO());
803    let domain_separator =
804        sample_vector_cbd_then_ntt::<K, ETA1, ETA1_RANDOMNESS_SIZE, Vector, Hasher>(
805            &mut r_as_ntt,
806            &prf_input,
807            0,
808        );
809    hax_lib::fstar!(
810        "eq_intro $randomness (Seq.slice $prf_input 0 32);
811        assert (v $domain_separator == v $K)"
812    );
813
814    // for i from 0 to k‚àí1 do
815    //     e1[i] := CBD_{Œ∑2}(PRF(r,N))
816    //     N := N + 1
817    // end for
818    let mut error_1 = from_fn(|_i| PolynomialRingElement::<Vector>::ZERO());
819    let domain_separator = sample_ring_element_cbd::<K, ETA2_RANDOMNESS_SIZE, ETA2, Vector, Hasher>(
820        &prf_input,
821        domain_separator,
822        &mut error_1,
823    );
824
825    // e_2 := CBD{Œ∑2}(PRF(r, N))
826    prf_input[32] = domain_separator;
827    hax_lib::fstar!(
828        "assert (Seq.equal $prf_input (Seq.append $randomness (Seq.create 1 $domain_separator)));
829        assert ($prf_input == Seq.append $randomness (Seq.create 1 $domain_separator))"
830    );
831    let prf_output: [u8; ETA2_RANDOMNESS_SIZE] = Hasher::PRF(&prf_input);
832    let error_2 = sample_from_binomial_distribution::<ETA2, Vector>(&prf_output);
833
834    // u := NTT^{-1}(AÀÜT ‚ó¶ rÀÜ) + e_1
835    let u = compute_vector_u(matrix, &r_as_ntt, &error_1);
836
837    // c_1 := Encode_{du}(Compress_q(u,d_u))
838    compress_then_serialize_u::<K, C1_LEN, U_COMPRESSION_FACTOR, BLOCK_LEN, Vector>(u, ciphertext);
839
840    (r_as_ntt, error_2)
841}

843#[inline(always)]
844pub(crate) fn encrypt_c2<
845    const K: usize,
846    const V_COMPRESSION_FACTOR: usize,
847    const C2_LEN: usize,
848    Vector: Operations,
849>(
850    t_as_ntt: &[PolynomialRingElement<Vector>; K],
851    r_as_ntt: &[PolynomialRingElement<Vector>; K],
852    error_2: &PolynomialRingElement<Vector>,
853    message: &[u8; SHARED_SECRET_SIZE],
854    ciphertext: &mut [u8],
855) {
856    // v := NTT^{‚àí1}(tÀÜT ‚ó¶ rÀÜ) + e_2 + Decompress_q(Decode_1(m),1)
857    let message_as_ring_element = deserialize_then_decompress_message(message);
858    let v = compute_ring_element_v(t_as_ntt, r_as_ntt, error_2, &message_as_ring_element);
859    hax_lib::fstar!("assert ($C2_LEN = Spec.MLKEM.v_C2_SIZE v_K)");

1089/// This function implements <strong>Algorithm 14</strong> of the
1090/// NIST FIPS 203 specification; this is the Kyber CPA-PKE decryption algorithm.
1091///
1092/// Algorithm 14 is reproduced below:
1093///
1094/// ```plaintext
1095/// Input: decryption key dk‚Çö‚Çñ‚Çë ‚àà ùîπ^{384k}.
1096/// Input: ciphertext c ‚àà ùîπ^{32(d·µ§k + d·µ•)}.
1097/// Output: message m ‚àà ùîπ^{32}.
1098///
1099/// c‚ÇÅ ‚Üê c[0 : 32d·µ§k]
1100/// c‚ÇÇ ‚Üê c[32d·µ§k : 32(d·µ§k + d·µ•)]
1101/// u ‚Üê Decompress_{d·µ§}(ByteDecode_{d·µ§}(c‚ÇÅ))
1102/// v ‚Üê Decompress_{d·µ•}(ByteDecode_{d·µ•}(c‚ÇÇ))
1103/// ≈ù ‚Üê ByteDecode‚ÇÅ‚ÇÇ(dk‚Çö‚Çñ‚Çë)
1104/// w ‚Üê v - NTT-¬π(≈ù·µÄ ‚ó¶ NTT(u))
1105/// m ‚Üê ByteEncode‚ÇÅ(Compress‚ÇÅ(w))
1106/// return m
1107/// ```
1108///
1109/// The NIST FIPS 203 standard can be found at
1110/// <https://csrc.nist.gov/pubs/fips/203/ipd>.
1111#[allow(non_snake_case)]
1112#[hax_lib::requires(fstar!(r#"Spec.MLKEM.is_rank $K /\
1113    $CIPHERTEXT_SIZE == Spec.MLKEM.v_CPA_CIPHERTEXT_SIZE $K /\
1114    $U_COMPRESSION_FACTOR == Spec.MLKEM.v_VECTOR_U_COMPRESSION_FACTOR $K /\
1115    $V_COMPRESSION_FACTOR == Spec.MLKEM.v_VECTOR_V_COMPRESSION_FACTOR $K /\
1116    $VECTOR_U_ENCODED_SIZE == Spec.MLKEM.v_C1_SIZE $K"#))]
1117#[hax_lib::ensures(|result|
1118    fstar!(r#"$result == Spec.MLKEM.ind_cpa_decrypt_unpacked $K $ciphertext
1119        (Libcrux_ml_kem.Polynomial.to_spec_vector_t #$K #$:Vector ${secret_key}.f_secret_as_ntt)"#)
1120)]
1121#[inline(always)]
1122pub(crate) fn decrypt_unpacked<
1123    const K: usize,
1124    const CIPHERTEXT_SIZE: usize,
1125    const VECTOR_U_ENCODED_SIZE: usize,
1126    const U_COMPRESSION_FACTOR: usize,
1127    const V_COMPRESSION_FACTOR: usize,
1128    Vector: Operations,
1129>(
1130    secret_key: &IndCpaPrivateKeyUnpacked<K, Vector>,
1131    ciphertext: &[u8; CIPHERTEXT_SIZE],
1132) -> [u8; SHARED_SECRET_SIZE] {
1133    // u := Decompress_q(Decode_{d_u}(c), d_u)
1134    let u_as_ntt = deserialize_then_decompress_u::<K, CIPHERTEXT_SIZE, U_COMPRESSION_FACTOR, Vector>(
1135        ciphertext,
1136    );
1137
1138    // v := Decompress_q(Decode_{d_v}(c + d_u¬∑k¬∑n / 8), d_v)
1139    let v = deserialize_then_decompress_ring_element_v::<K, V_COMPRESSION_FACTOR, Vector>(
1140        &ciphertext[VECTOR_U_ENCODED_SIZE..],
1141    );
1142
1143    // m := Encode_1(Compress_q(v ‚àí NTT^{‚àí1}(sÀÜT ‚ó¶ NTT(u)) , 1))
1144    let message = compute_message(&v, &secret_key.secret_as_ntt, &u_as_ntt);
1145    compress_then_serialize_message(message)
1146}
