13/// Return 1 if `value` is not zero and 0 otherwise.
14#[inline(never)] // Don't inline this to avoid that the compiler optimizes this out.
15#[hax_lib::ensures(|result| if value == 0 {result == 0} else {result == 1})]
16fn inz(value: u8) -> u8 {
17    #[cfg(hax)]
18    let _orig_value = value; // copy original value for proofs
19
20    let value = value as u16;
21    let result = ((!value).wrapping_add(1) >> 8) as u8;
22    let res = result & 1;
23
24    hax_lib::fstar!(
25        r#"lognot_lemma $value;
26           logand_lemma (mk_u8 1) $result"#
27    );
28    res
29}

31#[inline(never)] // Don't inline this to avoid that the compiler optimizes this out.
32#[hax_lib::ensures(|result| if value == 0 {result == 0} else {result == 1})]
33fn is_non_zero(value: u8) -> u8 {
34    #[cfg(eurydice)]
35    return inz(value);
36
37    #[cfg(not(eurydice))]
38    core::hint::black_box(inz(value))
39}

41/// Return 1 if the bytes of `lhs` and `rhs` do not exactly
42/// match and 0 otherwise.
43#[inline(never)] // Don't inline this to avoid that the compiler optimizes this out.
44#[hax_lib::requires(lhs.len() == rhs.len())]
45#[hax_lib::ensures(|result| if lhs == rhs {result == 0} else {result == 1})]
46fn compare(lhs: &[u8], rhs: &[u8]) -> u8 {
47    let mut r: u8 = 0;
48    for i in 0..lhs.len() {
49        hax_lib::loop_invariant!(|i: usize| {
50            fstar!(
51                r#"v $i <= Seq.length $lhs /\
52            (if (Seq.slice $lhs 0 (v $i) = Seq.slice $rhs 0 (v $i)) then
53                $r == (mk_u8 0)
54                else ~ ($r == (mk_u8 0)))"#
55            )
56        });
57
58        let nr = r | (lhs[i] ^ rhs[i]);
59
60        hax_lib::fstar!(
61            r#"if $r =. (mk_u8 0) then (
62            if (Seq.index $lhs (v $i) = Seq.index $rhs (v $i)) then (
63               logxor_lemma (Seq.index $lhs (v $i)) (Seq.index $rhs (v $i));
64               assert (((${lhs}.[ $i ] <: u8) ^. (${rhs}.[ $i ] <: u8) <: u8) = zero);
65               logor_lemma $r ((${lhs}.[ $i ] <: u8) ^. (${rhs}.[ $i ] <: u8) <: u8);
66               assert ($nr = $r);
67               assert (forall j. Seq.index (Seq.slice $lhs 0 (v $i)) j == Seq.index $lhs j);
68               assert (forall j. Seq.index (Seq.slice $rhs 0 (v $i)) j == Seq.index $rhs j);
69               eq_intro (Seq.slice $lhs 0 ((v $i) + 1)) (Seq.slice $rhs 0 ((v $i) + 1))
70            )
71            else (
72               logxor_lemma (Seq.index $lhs (v $i)) (Seq.index $rhs (v $i));
73               assert (((${lhs}.[ $i ] <: u8) ^. (${rhs}.[ $i ] <: u8) <: u8) <>  zero);
74               logor_lemma r ((${lhs}.[ $i ] <: u8) ^. (${rhs}.[ $i ] <: u8) <: u8);
75               assert (v $nr > 0);
76               assert (Seq.index (Seq.slice $lhs 0 ((v $i)+1)) (v $i) <> 
77                       Seq.index (Seq.slice $rhs 0 ((v $i)+1)) (v $i));
78               assert (Seq.slice $lhs 0 ((v $i)+1) <> Seq.slice $rhs 0 ((v $i) + 1))
79            )
80          ) else (
81            logor_lemma $r ((${lhs}.[ $i ] <: u8) ^. (${rhs}.[ $i ] <: u8) <: u8);
82            assert (v $nr >= v $r);
83            assert (Seq.slice $lhs 0 (v $i) <> Seq.slice $rhs 0 (v $i));
84            if (Seq.slice $lhs 0 ((v $i)+1) = Seq.slice $rhs 0 ((v $i) + 1)) then
85              (assert (forall j. j < (v $i) + 1 ==> Seq.index (Seq.slice $lhs 0 ((v $i)+1)) j == Seq.index (Seq.slice $rhs 0 ((v $i)+1)) j);
86               eq_intro (Seq.slice $lhs 0 (v $i)) (Seq.slice $rhs 0 (v $i));
87               assert(False))
88          )"#
89        );
90
91        r = nr;
92    }
93
94    is_non_zero(r)
95}

101    #[cfg(eurydice)]
102    return compare(lhs, rhs);

104    #[cfg(not(eurydice))]
105    core::hint::black_box(compare(lhs, rhs))

108/// If `selector` is not zero, return the bytes in `rhs`; return the bytes in
109/// `lhs` otherwise.
110#[inline(never)] // Don't inline this to avoid that the compiler optimizes this out.
111#[hax_lib::fstar::options("--ifuel 0 --z3rlimit 50")]
112#[hax_lib::requires(
113    lhs.len() == rhs.len() &&
114    lhs.len() == SHARED_SECRET_SIZE
115)]
116#[hax_lib::ensures(|result| if selector == 0 {result == lhs} else {result == rhs})]
117fn select_ct(lhs: &[u8], rhs: &[u8], selector: u8) -> [u8; SHARED_SECRET_SIZE] {
118    let mask = is_non_zero(selector).wrapping_sub(1);
119    hax_lib::fstar!(
120        "assert (if $selector = (mk_u8 0) then $mask = ones else $mask = zero);
121        lognot_lemma $mask;
122        assert (if $selector = (mk_u8 0) then ~.$mask = zero else ~.$mask = ones)"
123    );
124    let mut out = [0u8; SHARED_SECRET_SIZE];
125
126    for i in 0..SHARED_SECRET_SIZE {
127        hax_lib::loop_invariant!(|i: usize| {
128            fstar!(
129                r#"v $i <= v $SHARED_SECRET_SIZE /\
130            (forall j. j < v $i ==> (if ($selector =. (mk_u8 0)) then Seq.index $out j == Seq.index $lhs j else Seq.index $out j == Seq.index $rhs j)) /\
131            (forall j. j >= v $i ==> Seq.index $out j == (mk_u8 0))"#
132            )
133        });
134        hax_lib::fstar!(r#"assert ((${out}.[ $i ] <: u8) = (mk_u8 0))"#);
135        let outi = (lhs[i] & mask) | (rhs[i] & !mask);
136        hax_lib::fstar!(
137            r#"if ($selector = (mk_u8 0)) then (
138            logand_lemma (${lhs}.[ $i ] <: u8) $mask;
139            assert (((${lhs}.[ $i ] <: u8) &. $mask <: u8) == (${lhs}.[ $i ] <: u8));
140            logand_lemma (${rhs}.[ $i ] <: u8) (~.$mask);
141            assert (((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8) == zero);
142            logor_lemma ((${lhs}.[ $i ] <: u8) &. $mask <: u8) ((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8);
143            assert ((((${lhs}.[ $i ] <: u8) &. $mask <: u8) |. ((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8) <: u8) == (${lhs}.[ $i ] <: u8));
144            logor_lemma (${out}.[ $i ] <: u8) (${lhs}.[ $i ] <: u8);
145            assert (((${out}.[ $i ] <: u8) |. (((${lhs}.[ $i ] <: u8) &. $mask <: u8) |. ((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8) <: u8) <: u8) == (${lhs}.[ $i ] <: u8));
146            assert ($outi = (${lhs}.[ $i ] <: u8))
147          )
148          else (
149            logand_lemma (${lhs}.[ $i ] <: u8) $mask;
150            assert (((${lhs}.[ $i ] <: u8) &. $mask <: u8) == zero);
151            logand_lemma (${rhs}.[ $i ] <: u8) (~.$mask);
152            assert (((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8) == (${rhs}.[ $i ] <: u8));
153            logor_lemma (${rhs}.[ $i ] <: u8) zero;
154            assert ((logor zero (${rhs}.[ $i ] <: u8)) == (${rhs}.[ $i ] <: u8));
155            assert ((((${lhs}.[ $i ] <: u8) &. $mask <: u8) |. ((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8)) == (${rhs}.[ $i ] <: u8));
156            logor_lemma (${out}.[ $i ] <: u8) (${rhs}.[ $i ] <: u8);
157            assert (((${out}.[ $i ] <: u8) |. (((${lhs}.[ $i ] <: u8) &. $mask <: u8) |. ((${rhs}.[ $i ] <: u8) &. (~.$mask <: u8) <: u8) <: u8) <: u8) == (${rhs}.[ $i ] <: u8));
158            assert ($outi = (${rhs}.[ $i ] <: u8))
159          )"#
160        );
161        out[i] = outi;
162    }
163
164    hax_lib::fstar!(
165        "if ($selector =. (mk_u8 0)) then (
166            eq_intro $out $lhs
167        )
168        else (
169            eq_intro $out $rhs
170        )"
171    );
172    out
173}

180#[hax_lib::ensures(|result| if selector == 0 {result == lhs} else {result == rhs})]
181pub(crate) fn select_shared_secret_in_constant_time(
182    lhs: &[u8],
183    rhs: &[u8],
184    selector: u8,
185) -> [u8; SHARED_SECRET_SIZE] {
186    #[cfg(eurydice)]
187    return select_ct(lhs, rhs, selector);

190    core::hint::black_box(select_ct(lhs, rhs, selector))
