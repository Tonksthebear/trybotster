195fn ntt_layer_int_vec_step<Vector: Operations>(
196    mut a: Vector,
197    mut b: Vector,
198    zeta_r: i16,
199) -> (Vector, Vector) {
200    let t = Vector::montgomery_multiply_by_constant(b, zeta_r);
201    b = Vector::sub(a, &t);
202    a = Vector::add(a, &t);
203    (a, b)
204}

218pub(crate) fn ntt_at_layer_4_plus<Vector: Operations>(
219    zeta_i: &mut usize,
220    re: &mut PolynomialRingElement<Vector>,
221    layer: usize,
222    _initial_coefficient_bound: usize, // This can be used for specifying the range of values allowed in re
223) {
224    let step = 1 << layer;
225
226    let _zeta_i_init = *zeta_i;
227    for round in 0..(128 >> layer) {
228        *zeta_i += 1;
229
230        let offset = round * step * 2;
231        let offset_vec = offset / 16; //FIELD_ELEMENTS_IN_VECTOR;
232        let step_vec = step / 16; //FIELD_ELEMENTS_IN_VECTOR;
233
234        for j in offset_vec..offset_vec + step_vec {
235            let (x, y) = ntt_layer_int_vec_step(
236                re.coefficients[j],
237                re.coefficients[j + step_vec],
238                zeta(*zeta_i),
239            );
240            re.coefficients[j] = x;
241            re.coefficients[j + step_vec] = y;
242        }
243    }
244}

270pub(crate) fn ntt_at_layer_7<Vector: Operations>(re: &mut PolynomialRingElement<Vector>) {
271    let step = VECTORS_IN_RING_ELEMENT / 2;
272    hax_lib::fstar!(r#"assert (v $step == 8)"#);
273    for j in 0..step {
274        hax_lib::loop_invariant!(|j: usize| {
275            fstar!(
276                r#"(v j < 8 ==>
277          (forall (i:nat). (i >= v j /\ i < 8) ==>
278            ntt_layer_7_pre (re.f_coefficients.[ sz i ]) (re.f_coefficients.[ sz i +! sz 8 ])))"#
279            )
280        });
281        hax_lib::fstar!(r#"reveal_opaque (`%ntt_layer_7_pre) (ntt_layer_7_pre #$:Vector)"#);
282        let t = Vector::multiply_by_constant(re.coefficients[j + step], -1600);
283        re.coefficients[j + step] = Vector::sub(re.coefficients[j], &t);
284        re.coefficients[j] = Vector::add(re.coefficients[j], &t);
285    }
286}

297pub(crate) fn ntt_binomially_sampled_ring_element<Vector: Operations>(
298    re: &mut PolynomialRingElement<Vector>,
299) {
300    // Due to the small coefficient bound, we can skip the first round of
301    // Montgomery reductions.
302    ntt_at_layer_7(re);
303
304    let mut zeta_i = 1;
305    ntt_at_layer_4_plus(&mut zeta_i, re, 6, 11207);
306    ntt_at_layer_4_plus(&mut zeta_i, re, 5, 11207 + 3328);
307    ntt_at_layer_4_plus(&mut zeta_i, re, 4, 11207 + 2 * 3328);
308    ntt_at_layer_3(&mut zeta_i, re, 11207 + 3 * 3328);
309    ntt_at_layer_2(&mut zeta_i, re, 11207 + 4 * 3328);
310    ntt_at_layer_1(&mut zeta_i, re, 11207 + 5 * 3328);
311
312    re.poly_barrett_reduce()
313}
