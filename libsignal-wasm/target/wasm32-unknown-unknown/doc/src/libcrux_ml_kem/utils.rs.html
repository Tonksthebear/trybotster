6/// Pad the `slice` with `0`s at the end.
7#[inline(always)]
8#[cfg_attr(hax, hax_lib::requires(
9    slice.len() <= LEN
10))]
11#[cfg_attr(hax, hax_lib::ensures(|result|
12    fstar!(r#"$result == Seq.append $slice (Seq.create (v $LEN - v (${slice.len()})) (mk_u8 0))"#)))]
13pub(crate) fn into_padded_array<const LEN: usize>(slice: &[u8]) -> [u8; LEN] {
14    let mut out = [0u8; LEN];
15    out[0..slice.len()].copy_from_slice(slice);
16    hax_lib::fstar!(r#"assert (Seq.slice out 0 (Seq.length slice) == slice)"#);
17    hax_lib::fstar!(
18        r#"assert (Seq.slice out (Seq.length slice) (v v_LEN) == Seq.slice (Seq.create (v v_LEN) (mk_u8 0)) (Seq.length slice) (v v_LEN))"#
19    );
20    hax_lib::fstar!(
21        "assert (forall i. i < Seq.length slice ==> Seq.index out i == Seq.index slice i)"
22    );
23    hax_lib::fstar!(
24        r#"assert (forall i. (i >= Seq.length slice && i < v v_LEN) ==> Seq.index out i == Seq.index (Seq.slice out (Seq.length slice) (v v_LEN)) (i - Seq.length slice))"#
25    );
26    hax_lib::fstar!(
27        "Seq.lemma_eq_intro out (Seq.append slice (Seq.create (v v_LEN - Seq.length slice) (mk_u8 0)))"
28    );
29    out
30}

32#[inline(always)]
33#[hax_lib::fstar::options("--z3rlimit 200")]
34#[hax_lib::requires(fstar!(r#"range (v $domain_separator + v $K) u8_inttype"#))]
35#[hax_lib::ensures(|ds|
36    fstar!(r#"v $ds == v $domain_separator + v $K /\
37            (forall (i:nat). i < v $K ==>
38                v (Seq.index (Seq.index ${prf_inputs}_future i) 32) == v $domain_separator + i /\
39                Seq.slice (Seq.index ${prf_inputs}_future i) 0 32 == Seq.slice (Seq.index $prf_inputs i) 0 32)"#)
40)]
41pub(crate) fn prf_input_inc<const K: usize>(
42    prf_inputs: &mut [[u8; 33]; K],
43    mut domain_separator: u8,
44) -> u8 {
45    #[cfg(hax)]
46    let _domain_separator_init = domain_separator;
47    #[cfg(hax)]
48    let _prf_inputs_init = prf_inputs.clone();
49
50    for i in 0..K {
51        hax_lib::loop_invariant!(|i: usize| {
52            fstar!(
53                r#"v $domain_separator == v $_domain_separator_init + v $i /\
54          (v $i < v $K ==> (forall (j:nat). (j >= v $i /\ j < v $K) ==>
55            prf_inputs.[ sz j ] == ${_prf_inputs_init}.[ sz j ])) /\
56          (forall (j:nat). j < v $i ==> v (Seq.index (Seq.index prf_inputs j) 32) == v $_domain_separator_init + j /\
57            Seq.slice (Seq.index prf_inputs j) 0 32 == Seq.slice (Seq.index $_prf_inputs_init j) 0 32)"#
58            )
59        });
60        prf_inputs[i][32] = domain_separator;
61        domain_separator += 1;
62    }
63    domain_separator
64}
